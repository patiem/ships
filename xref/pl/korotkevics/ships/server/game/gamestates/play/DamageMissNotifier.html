8   /**
9    * Notification for client that his shot missed a target.
10   *
11   * @author Piotr Czy≈º
12   * @since 2018-01-02
13   */
14  class DamageMissNotifier extends DamageNotifier {
15    DamageMissNotifier(final MessageSender messageSender, final TurnManager turnManager) {
16      super(messageSender, turnManager);
17      this.header = Header.MISS;
18    }
19  
20    @Override
21    protected void notify(final Message receivedShot) {
22      messageSender.send(turnManager.getCurrentPlayer(), header);
23      informOpponentAboutShot(receivedShot);
24      this.turnManager.switchPlayer();
25    }
26  }
