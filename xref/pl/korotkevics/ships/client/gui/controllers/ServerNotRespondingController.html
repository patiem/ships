17  /**
18   * Server Not Responding window controller.
19   *
20   * @author Magdalena Aarsman
21   * @since 2017-12-17
22   */
23  
24  public class ServerNotRespondingController implements Initializable {
25  
26    private static final Target logger = new SharedLogger(Client.class);
27  
28    @FXML
29    private AnchorPane mainAnchorPane;
30    private ResourceBundle resourceBundle;
31  
32    @FXML
33    private void onTryAgainPressed() {
34      try {
35        final String connectWindowUrl = "/fxml/connectWindow.fxml";
36        final FXMLLoader connectLoader = new FXMLLoader(getClass().getResource(connectWindowUrl));
37        connectLoader.setResources(this.resourceBundle);
38        final Parent connectWindow = connectLoader.load();
39        final Pane mainPane = (Pane) mainAnchorPane.getParent();
40        mainPane.getChildren().clear();
41        mainPane.getChildren().setAll(connectWindow);
42      } catch (IOException e) {
43        logger.error(e.getMessage());
44      }
45    }
46  
47    @Override
48    public void initialize(final URL location, final ResourceBundle resources) {
49      this.resourceBundle = resources;
50    }
51  }
