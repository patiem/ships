163   private void initializeBoard() {
164     final int margin = 50;
165     final NumberBinding allRectanglesWidth = Bindings.min(yourBoard.widthProperty(),
166         yourBoard.widthProperty().add(-margin));
167     final NumberBinding allRectanglesHeight = Bindings.min(yourBoard.heightProperty(),
168         yourBoard.widthProperty()).add(-margin);
169 
170     for (int i = 0; i < BOARD_SIZE; i++) {
171       for (int j = 0; j < BOARD_SIZE; j++) {
172         final int recIndex = j * BOARD_SIZE + i;
173         final int fillIndex = i * BOARD_SIZE + j;
174 
175         Rectangle yourRect = getYourRect(allRectanglesWidth, allRectanglesHeight);
176         yourBoard.add(yourRect, i, j);
177         setEventsOnField(yourRect, recIndex, fillIndex);
178         GridPane.setHalignment(yourRect, HPos.CENTER);
179       }
180     }
181   }

183   private Rectangle getYourRect(NumberBinding allRectanglesWidth,
184                                 NumberBinding allRectanglesHeight) {
185     final int initialSize = 15;
186     Rectangle yourRect = new Rectangle(initialSize, initialSize, Color.GRAY);
187     yourRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
188     yourRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
189     return yourRect;
190   }

199   private void setOpacity(int index, int mastCount, double opacity) {
200     index += 1;
201 
202     if (shipOrientation.equals(ShipOrientation.HORIZONTAL)) {
203       for (int i = 1; i < mastCount; i++) {
204         (yourBoard.getChildren().get(index + i * BOARD_SIZE)).setOpacity(opacity);
205       }
206     } else {
207       for (int i = 1; i < mastCount; i++) {
208         (yourBoard.getChildren().get(index + i)).setOpacity(opacity);
209       }
210     }
211   }

213   private void setFieldOnDragExited(Rectangle rectangle, int fillIndex) {
214     rectangle.setOnDragExited(
215         event -> {
216           double noOpacity = 1.0;
217           int mastCount = ((Group) event.getGestureSource()).getChildren().size();
218           if (isOutOfBound(fillIndex, mastCount)) {
219             return;
220           }
221           ((Rectangle) event.getSource()).setOpacity(noOpacity);
222           setOpacity(fillIndex, mastCount ,noOpacity);
223           event.consume();
224         });
225   }
226 
227   private void setFieldOnDragEntered(Rectangle rectangle, final int index) {
228     rectangle.setOnDragEntered(event -> {
229       double opacity = 0.5;
230       int mastCount = ((Group) event.getGestureSource()).getChildren().size();
231       if (shipOrientation.equals(ShipOrientation.HORIZONTAL) && isOutOfBound(index, mastCount)) {
232         return;
233       }
234       GridToBoardConverter gridToBoardConverter = new GridToBoardConverter(yourBoard);
235       ShipPlacementValidator shipPlacementValidator =
236           new ShipPlacementValidator(shipOrientation, index, mastCount,
237               gridToBoardConverter.convert());
238       if (!shipPlacementValidator.isPlacementValid()) {
239         return;
240       }
241 
242       ((Rectangle) event.getSource()).setOpacity(opacity);
243       setOpacity(index, mastCount, opacity);
244       event.consume();
245     });
246   }

248   private boolean isOutOfBound(int index, int mastCount) {
249     final int notRectangleChildCount = 2;
250     if (shipOrientation.equals(ShipOrientation.HORIZONTAL)) {
251       return index + (mastCount - 1) * BOARD_SIZE
252           > yourBoard.getChildren().size() - notRectangleChildCount;
253     } else {
254       if (index + mastCount >= yourBoard.getChildren().size()) {
255         return true;
256       }
257     }
258     return false;
259   }

261   private void setFieldOnDragDropped(Rectangle rectangle, final int recIndex, final int index) {
262     rectangle.setOnDragDropped(event -> {
263       Dragboard db = event.getDragboard();
264       boolean success = false;
265       if (db.hasString()) {
266         success = true;
267       }
268       event.setDropCompleted(success);
269       int mastCount = ((Group) event.getGestureSource()).getChildren().size();
270       Mast[] masts = new Mast[mastCount];
271 
272       if (shipOrientation.equals(ShipOrientation.HORIZONTAL) && isOutOfBound(index, mastCount)) {
273         shipPlacementSuccess = false;
274         return;
275       }
276 
277       GridToBoardConverter gridToBoardConverter = new GridToBoardConverter(yourBoard);
278       ShipPlacementValidator shipPlacementValidator =
279           new ShipPlacementValidator(shipOrientation, index, mastCount,
280               gridToBoardConverter.convert());
281       if (!shipPlacementValidator.isPlacementValid()) {
282         shipPlacementSuccess = false;
283         return;
284       }
285       masts[0] = Mast.ofIndex(String.valueOf(recIndex));
286       rectangle.setFill(Color.GREEN);
287       placeShip(index, masts, mastCount, recIndex);
288       ships.add(Ship.ofMasts(masts));
289       if (ships.size() == SHIPS_COUNT) {
290         buttonReady.setDisable(false);
291         fleet = Fleet.ofShips(ships);
292       }
293       shipPlacementSuccess = true;
294       event.consume();
295     });
296   }

298   private void placeShip(int index, Mast[] masts, int mastCount, int recIndex) {
299     index += 1;
300 
301     if (shipOrientation.equals(ShipOrientation.VERTICAL)) {
302       for (int i1 = 1; i1 < mastCount; i1++) {
303         ((Rectangle) yourBoard.getChildren().get(index + i1)).setFill(Color.GREEN);
304         masts[i1] = Mast.ofIndex(String.valueOf(recIndex + i1 * BOARD_SIZE));
305       }
306     } else {
307       for (int i1 = 1; i1 < mastCount; i1++) {
308         ((Rectangle) yourBoard.getChildren().get(index + i1 * BOARD_SIZE)).setFill(Color.GREEN);
309         masts[i1] = Mast.ofIndex(String.valueOf(recIndex + i1));
310       }
311     }
312   }

339   private void loadGameWindow() {
340     try {
341       this.getClient().sendFleet(fleet);
342 
343       final String gameWindowUrl = "/fxml/gameWindow.fxml";
344       final FXMLLoader gameWindowLoader = new FXMLLoader(getClass().getResource(gameWindowUrl));
345       gameWindowLoader.setResources(this.resourceBundle);
346       final Parent gameWindow = gameWindowLoader.load();
347       final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
348 
349       Rectangle2D screenBounds = Screen.getPrimary().getVisualBounds();
350 
351       Stage stage = (Stage) mainPane.getScene().getWindow();
352       final int shrinkSize = 300;
353       stage.setMinHeight(screenBounds.getHeight() - shrinkSize);
354       stage.setMinWidth(screenBounds.getWidth() - shrinkSize);
355 
356       mainPane.getChildren().clear();
357       mainPane.getChildren().setAll(gameWindow);
358 
359       GameWindowController gameController = gameWindowLoader.getController();
360       gameController.initializeClient();
361       gameController.initializeBoards(yourBoard);
362 
363       final double margin = 0.0;
364       AnchorPane.setTopAnchor(gameWindow, margin);
365       AnchorPane.setBottomAnchor(gameWindow, margin);
366       AnchorPane.setLeftAnchor(gameWindow, margin);
367       AnchorPane.setRightAnchor(gameWindow, margin);
368 
369     } catch (IOException e) {
370       logger.error(e.getMessage());
371     }
372   }

374   private void askForRandomFleet() {
375     final ButtonType yesButton = new ButtonType(this.resourceBundle.getString("yes"), ButtonBar.ButtonData.YES);
376     final ButtonType noButton = new ButtonType(this.resourceBundle.getString("no"), ButtonBar.ButtonData.NO);
377 
378     Alert alert = showInfoAlert(yesButton, noButton);
379     if (alert.getResult() == noButton) {
380       return;
381     }
382 
383     this.disableDragAndDropShips(true);
384     this.clearBoard();
385     this.buttonRandom.setDisable(true);
386     this.getClient().askForRandomFleet();
387   }

389   private Alert showInfoAlert(final ButtonType yesButton,final ButtonType noButton) {
390     final Alert alert = new Alert(Alert.AlertType.INFORMATION, this.resourceBundle.getString("alertInfo")
391         + System.lineSeparator()
392         + System.lineSeparator()
393         + this.resourceBundle.getString("alertQuestion"),
394         yesButton, noButton);
395     alert.getDialogPane().setMinHeight(Region.USE_PREF_SIZE);
396     alert.setTitle(this.resourceBundle.getString("information"));
397     alert.setHeaderText(this.resourceBundle.getString("information"));
398     alert.showAndWait();
399     return alert;
400   }

409   private void disableDragAndDropForGroupOfShips(final Group shipsGroup, final boolean disable) {
410     for (Node ship : shipsGroup.getChildren()) {
411       ship.setDisable(disable);
412       for (Node child : ((Group) ship).getChildren()) {
413         final Rectangle rec = (Rectangle) child;
414         if(disable) {
415           rec.setFill(Color.GRAY);
416         } else {
417           final String purple = "#7A16C2";
418           rec.setFill(Color.web(purple));
419         }
420       }
421     }
422   }

442   private int convertToGridIndex(final int index) {
443     final int column = index / BOARD_SIZE;
444     final int row = index - (column * BOARD_SIZE);
445     return row * BOARD_SIZE + column + 1;
446   }

461   @Override
462   public void initialize(final URL location, final ResourceBundle resources) {
463     this.resourceBundle = resources;
464     buttonReady.setOnAction(event -> loadGameWindow());
465     buttonReady.setDisable(true);
466     initializeBoard();
467     addDragEventsToShips();
468     ships = new ArrayList<>(SHIPS_COUNT);
469     shipOrientation = ShipOrientation.VERTICAL;
470     rbVertical.setOnAction(event -> shipOrientation = ShipOrientation.VERTICAL);
471     rbHorizontal.setOnAction(event -> shipOrientation = ShipOrientation.HORIZONTAL);
472     buttonRandom.setOnAction(event -> askForRandomFleet());
473     eventButton.addEventHandler(RandomPlacementEvent.RANDOM_PLACEMENT_EVENT,
474         event -> getRandomFleet(event.getFleet()));
475     buttonClear.setOnAction(event -> clearBoardAction());
476     this.addDropShadows();
477   }

479   private void addDropShadows() {
480     final DropShadow shadow = new DropShadow();
481     this.buttonRandom.addEventHandler(MouseEvent.MOUSE_ENTERED, mouseEvent -> buttonRandom.setEffect(shadow));
482     this.buttonRandom.addEventHandler(MouseEvent.MOUSE_EXITED, mouseEvent -> buttonRandom.setEffect(null));
483     this.buttonReady.addEventHandler(MouseEvent.MOUSE_ENTERED, mouseEvent -> buttonReady.setEffect(shadow));
484     this.buttonReady.addEventHandler(MouseEvent.MOUSE_EXITED, mouseEvent -> buttonReady.setEffect(null));
485     this.buttonClear.addEventHandler(MouseEvent.MOUSE_ENTERED, mouseEvent -> buttonClear.setEffect(shadow));
486     this.buttonClear.addEventHandler(MouseEvent.MOUSE_EXITED, mouseEvent -> buttonClear.setEffect(null));
487   }
