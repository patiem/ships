24  /**
25   * Main window (containing title and similar) controller.
26   *
27   * @author Magdalena Aarsman, Sandor Korotkevics
28   * @since 2017-12-14
29   */
30  
31  public class MainController {
32    
33    private static final Target logger = new SharedLogger(Client.class);
34    
35    private static final String DICTIONARY = "dict";
36    
37    @FXML
38    private Label gameTitle;
39    
40    @FXML
41    private Button russianButton;
42    
43    @FXML
44    private Pane mainPane;
45    
46    @Getter
47    private Client client;
48    
49    @FXML
50    private Button polishButton;
51    
52    @FXML
53    private Button englishButton;
54    
55    private ResourceBundle resourceBundle;
56    
57    /**
58     * Initialize main controller.
59     *
60     * @param client
61     *     - Instance of Client, which communicate with server.
62     */
63    @FXML
64    public void initialize(final Client client) {
65      this.client = client;
66      this.loadDefaultView();
67  
68    }
69    
70    private void loadDefaultView() {
71      this.reloadView();
72    }
73    
74    private void reloadView() {
75      final FXMLLoader fxmlLoader = this.prepareFxmlLoader();
76      this.gameTitle.setText(this.resourceBundle.getString("gameTitle"));
77      try {
78        this.addToMainPane(fxmlLoader.load());
79      } catch (IOException e) {
80        logger.error(e.getMessage());
81      }
82    }
83    
84    private FXMLLoader prepareFxmlLoader() {
85      final String url = "/fxml/connectWindow.fxml";
86      final FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource(url));
87      this.resourceBundle = ResourceBundle.getBundle(DICTIONARY, new UTF8Control());
88      fxmlLoader.setResources(this.resourceBundle);
89      return fxmlLoader;
90    }
91    
92    private void addToMainPane(final Parent parent) {
93      this.mainPane.getChildren().clear();
94      this.mainPane.getChildren().add(parent);
95    }
96    
97    @FXML
98    void triggerPolishVersion(final ActionEvent event) {
99      logger.info("The language is set to " + OurLocale.POLISH);
100     Locale.setDefault(new Locale(OurLocale.POLISH.toString()));
101     this.reloadView();
102     this.changeTitleLabel();
103   }
104 
105   private void changeTitleLabel() {
106     Scene scene = mainPane.getScene();
107     Stage stage = (Stage) scene.getWindow();
108     stage.setTitle(this.resourceBundle.getString("gameTitle"));
109   }
110 
111   @FXML
112   void triggerEnglishVersion(final ActionEvent event) {
113     logger.info("The language is set to " + OurLocale.ENGLISH);
114     Locale.setDefault(new Locale(OurLocale.ENGLISH.toString()));
115     this.reloadView();
116     this.changeTitleLabel();
117   }
118   
119   @FXML
120   void triggerRussianVersion(final ActionEvent event) {
121     logger.info("The language is set to " + OurLocale.RUSSIAN);
122     Locale.setDefault(new Locale(OurLocale.RUSSIAN.toString()));
123     this.reloadView();
124     this.changeTitleLabel();
125   }
126   
127   void disableLocalizationButtons() {
128     this.hideButton(this.polishButton);
129     this.hideButton(this.englishButton);
130     this.hideButton(this.russianButton);
131   }
132   
133   private void hideButton(Button button) {
134     button.setDisable(true);
135     button.setVisible(false);
136   }
137 }
