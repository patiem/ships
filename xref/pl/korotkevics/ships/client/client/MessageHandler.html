3   import javafx.scene.control.Button;
4   import lombok.Getter;
5   import pl.korotkevics.ships.client.reporting.Reporter;
6   import pl.korotkevics.ships.shared.infra.communication.api.Message;
7   import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
8   import pl.korotkevics.ships.shared.infra.communication.api.message.Status;
9   import pl.korotkevics.ships.shared.infra.logging.api.Target;
10  import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
11  
12  import java.util.Map;
13  
14  /**
15   * Handling message, enable to react in proper way to server messages.
16   *
17   * @author Magdalena Aarsman
18   * @since 2017-12-16
19   */
20  class MessageHandler {
21  
22    private static final Target logger = new SharedLogger(Client.class);
23    
24    private final Map<Header, EventTrigger> triggers;
25    private Reporter reporter;
26    private Button eventButton;
27    @Getter
28    private boolean endConnectionTriggered;
29  
30    /**
31     * Creates an instance of MessageHandler and configures all possible triggers types.
32     */
33    MessageHandler(final Map<Header, EventTrigger> triggerMap, final Reporter reporter) {
34      this.triggers = triggerMap;
35      this.reporter = reporter;
36    }
37  
38    Button getCurrentEventButton() {
39      return eventButton;
40    }
41  
42    void setCurrentEventButton(Button eventButton) {
43      this.eventButton = eventButton;
44    }
45  
46    void handle(Message message) {
47      if (eventButton == null) {
48        throw new IllegalStateException("there is no object on which there can be fire event on");
49      }
50  
51      Header header = message.getHeader();
52  
53      if (!triggers.containsKey(header)) {
54        logger.error("message header: " + header + " is unknown");
55        return;
56      }
57      checkIfEndWillBeTriggered(message);
58      triggers.get(header).fire(eventButton, message);
59      if (this.reporter!=null) {
60        this.reporter.report(triggers.get(header).provideDescription());
61      }
62    }
63  
64    private void checkIfEndWillBeTriggered(final Message message) {
65      if (Status.END.equals(message.getStatus())
66          || Header.WIN.equals(message.getHeader())
67          || Header.LOSE.equals(message.getHeader())) {
68        endConnectionTriggered = true;
69      }
70    }
71  }
