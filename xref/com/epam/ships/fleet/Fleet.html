8   import java.util.Collections;

10  import java.util.List;
11  import java.util.Map;

13  /**
14   * Fleet of ships.
15   * @author Piotr Czy≈º, Sandor Korotkevics
16   * @see Ship
17   * @since 2017-12-19
18   */
19  @EqualsAndHashCode
20  @RequiredArgsConstructor(staticName = "fleet", access = AccessLevel.PRIVATE)
21  @ToString
22  public class Fleet {
23  
24    private final Map<Mast, Ship> fleet;
25  
26    private Fleet() {
27      this.fleet = Collections.emptyMap();
28    }
29  
30    /**
31     * Static factory method.
32     *
33     * @param ships - list of ships
34     * @return Fleet
35     */
36    public static Fleet ofShips(List<Ship> ships) {
37      Map<Mast, Ship> fleet = new HashMap<>();
38      ships.forEach(ship -> ship.getMasts().forEach(mast -> fleet.put(mast, ship)));
39      return new Fleet(fleet);
40    }
41  
42    public static Fleet empty() {
43      return new Fleet();
44    }
45  
46    /**
47     * @return true if a fleet is defeated.
48     */
49    public boolean isDefeated() {
50      return this.fleet.isEmpty();
51    }
52  
53    /**
54     * It handles an attack
55     * returning an output of such.
56     *
57     * @param mast attacked
58     * @return Damage caused
59     */
60    public Damage handleDamage(Mast mast) {
61      if (!this.fleet.containsKey(mast)) {
62        return Damage.MISSED;
63      }
64      Ship shipHit = this.fleet.get(mast);
65      this.fleet.remove(mast);
66      if (!this.fleet.containsValue(shipHit)) {
67        return Damage.DESTRUCTED;
68      }
69      return Damage.HIT;
70    }
71  }
