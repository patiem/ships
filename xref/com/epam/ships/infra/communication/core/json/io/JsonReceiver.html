17  /**
18   * It receives a message from an input stream.
19   * @author Piotr Czy≈º, Magda Aarsman, Sandor Korotkevics
20   * @see Receiver
21   * @see JsonReceiver
22   * @see Decoder
23   * @see JsonDecoder
24   * @see Message
25   * @since 2017-12-07
26   */
27  
28  public class JsonReceiver implements Receiver {
29  
30    private final InputStream inputStream;
31  
32    /**
33     * @param inputStream it will read from this stream.
34     */
35    public JsonReceiver(InputStream inputStream) {
36      this.inputStream = inputStream;
37    }
38  
39    /**
40     * It reads from input stream, coverts it first
41     * to a JsonElement and the to a BaseMessage.
42     * If there is nothing to read from an input stream,
43     * it returns an corresponding BaseMessage as well.
44     *
45     * @return BaseMessage input interpreted as a message.
46     */
47    @Override
48    public Message receive() {
49      Scanner scanner = new Scanner(inputStream, "UTF-8");
50      StringBuilder stringBuilder = new StringBuilder();
51      if (scanner.hasNextLine()) {
52        stringBuilder.append(scanner.nextLine());
53      } else {
54        return new MessageBuilder()
55            .withHeader(Header.CONNECTION)
56            .withStatus(Status.END)
57            .withStatement("End of a message")
58            .build();
59      }
60      Decoder<JsonElement> jsonDecoder = new JsonDecoder();
61      JsonParser jsonParser = new JsonParser();
62      JsonElement jsonElement = jsonParser.parse(stringBuilder.toString());
63      return jsonDecoder.decode(jsonElement);
64    }
65  }
