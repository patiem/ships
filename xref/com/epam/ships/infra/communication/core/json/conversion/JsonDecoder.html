3   import com.google.gson.Gson;
4   import com.google.gson.GsonBuilder;
5   import com.google.gson.JsonElement;
6   
7   import com.epam.ships.infra.communication.api.Message;
8   import com.epam.ships.infra.communication.api.conversion.Decoder;
9   import com.epam.ships.infra.communication.core.message.BaseMessage;
10  
11  /**
12   * It converts a JsonElement into a BaseMessage instance.
13   *
14   * @author Sandor Korotkevics
15   * @see Decoder
16   * @see Message
17   * @see com.epam.ships.infra.communication.api.Message
18   * @since 2017-12-10
19   */
20  public class JsonDecoder implements Decoder<JsonElement> {
21  
22    /**
23     * It converts a JsonElement instance into a BaseMessage
24     * instance.
25     *
26     * @param jsonElement a JsonElement instance representing
27     *                    a BaseMessage.
28     * @return BaseMessage a result of conversion of JsonElement into BaseMessage
29     */
30    @Override
31    public Message decode(JsonElement jsonElement) {
32      Gson gson = new GsonBuilder().create();
33  
34      return gson.fromJson(jsonElement, BaseMessage.class);
35    }
36  }
