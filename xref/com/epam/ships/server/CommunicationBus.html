1   package com.epam.ships.server;

3   import com.epam.ships.infra.communication.api.Message;
4   import com.epam.ships.infra.logging.api.Target;
5   import com.epam.ships.infra.logging.core.SharedLogger;

7   import java.io.IOException;
8   import java.net.Socket;
9   import java.util.ArrayList;
10  import java.util.List;
11  
12  /**
13   * It starts a server, gets its client sockets
14   * and adds them wrapped into own list.
15   * @author Piotr Czy≈º
16   * @since 2017-12-09
17   */
18  
19  class CommunicationBus {
20  
21    private static final int SERVER_PORT = 8189;
22    private final Target logger = new SharedLogger(CommunicationBus.class);
23    private final AppServer appServer;
24  
25    private List<WrappedClient> clients;
26  
27    CommunicationBus() throws IOException {
28      appServer = new AppServer(SERVER_PORT);
29      clients = new ArrayList<>();
30    }
31  
32    void start() throws IOException {
33      appServer.connectClients();
34      for (Socket socketClient : appServer.getClientSockets()) {
35        wrapClient(socketClient);
36      }
37      logger.info("Communication bus started..");
38    }
39  
40    private void wrapClient(Socket socketClient) {
41      WrappedClient wrappedClient = new WrappedClient(socketClient);
42      clients.add(wrappedClient);
43    }
44  
45    void stop() {
46      for (WrappedClient c : clients) {
47        c.close();
48      }
49  
50      clients.clear();
51    }
52  
53    Message receive(WrappedClient sender) {
54      return sender.receive();
55    }
56  
57    void send(final WrappedClient recipient, final Message response) {
58      recipient.send(response);
59    }
60  
61    final WrappedClient getFirstClient() {
62      return this.clients.get(0);
63    }
64  
65    final WrappedClient getSecondClient() {
66      return this.clients.get(1);
67    }
68  }
