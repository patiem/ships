020import java.util.*;
021
022import static org.mockito.Mockito.mock;
023import static org.mockito.Mockito.times;
024import static org.mockito.Mockito.verify;
025import static org.mockito.MockitoAnnotations.initMocks;
026import static org.testng.Assert.*;
027
028@Test
029public class ShotHandlerTest {
030
031
032  List<Fleet> fleets;
033  ShotHandler shotHandler;
034
035  @Mock
036  TurnManager turnManager;
037  @Mock
038  CommunicationBus communicationBus;
039
040  @BeforeMethod
041  void before(){
042    initMocks(this);
043    Fleet fleet = Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("3"), Mast.ofIndex("2"), Mast.ofIndex("1"))));
044    Fleet fleet2 = Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("11"), Mast.ofIndex("12"), Mast.ofIndex("13"))));
045    fleets = Arrays.asList(fleet,fleet2);
046    shotHandler = new ShotHandler(communicationBus, turnManager, fleets);
047  }
048
049  public void shouldHandleHitWhenFirstPlayerShots() {
050    //given
051    //when
052    Message message = shotAtIndex("11");
053    boolean result = shotHandler.handle(true, message);
054
055    //then
056    assertEquals(fleets.get(1).isDefeated(),result);
057    assertFalse(result);
058    verify(turnManager, times(0)).switchPlayer();
059  }
060
061  public void shouldHandleHitWhenSecondPlayerShots() {
062    //given
063    //when
064    Message message = shotAtIndex("3");
065    boolean result = shotHandler.handle(false, message);
066
067    //then
068    assertEquals(fleets.get(1).isDefeated(),result);
069    assertFalse(result);
070    verify(turnManager, times(0)).switchPlayer();
071  }
072
073
074  public void shouldHandleMiss() {
075    //given
076    //when
077    Message message = shotAtIndex("112");
078    boolean result = shotHandler.handle(true, message);
079
080    //then
081    assertEquals(fleets.get(1).isDefeated(),result);
082    assertFalse(result);
083    verify(turnManager, times(1)).switchPlayer();
084  }
085
086  public void shouldHandleFleetDefeat() {
087    //given
088    ShotHandler shotHandler = new ShotHandler(communicationBus, turnManager, fleets);
089
090
091    //when
092    Message message = shotAtIndex("11");
093    Message message2 = shotAtIndex("12");
094    Message message3 = shotAtIndex("13");
095    shotHandler.handle(true, message);
096    shotHandler.handle(true, message2);
097    boolean result = shotHandler.handle(true, message3);
098
099    //then
100    assertEquals(fleets.get(1).isDefeated(),result);
101    assertTrue(result);
102    verify(turnManager, times(0)).switchPlayer();
103  }
104
105  private Message shotAtIndex(String mast) {
106    return new MessageBuilder()
107        .withAuthor(Author.CLIENT)
108        .withHeader(Header.SHOT)
109        .withStatement(mast)
110        .build();
111  }
112
113}
