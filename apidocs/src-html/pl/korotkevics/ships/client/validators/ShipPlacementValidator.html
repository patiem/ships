001package pl.korotkevics.ships.client.validators;
002
003import pl.korotkevics.ships.client.gui.util.FieldState;
004import pl.korotkevics.ships.client.gui.util.ShipOrientation;
005import lombok.RequiredArgsConstructor;
006
007import java.util.stream.IntStream;
008
009/**
010 * Enable validation of ships placement.
011 * @author Magdalena Aarsman
012 * @since 2017-12-31
013 */
014@RequiredArgsConstructor
015public class ShipPlacementValidator {
016  private static final int BOARD_SIZE = 10;
017
018  private final ShipOrientation shipOrientation;
019  private final int shipStartIndex;
020  private final int mastCount;
021  private final FieldState[] board;
022
023  /**
024   * Check if ship placement is valid.
025   * @return true if it is valid, false if not.
026   */
027  public boolean isPlacementValid() {
028    return !this.isOutOfRange()
029        && (shipOrientation.equals(ShipOrientation.HORIZONTAL)
030        ? this.checkForHorizontal()
031        : this.checkForVertical());
032  }
033
034  private boolean checkIfBreakingThroughBorders() {
035    int startIndexInColumn = shipStartIndex % BOARD_SIZE;
036    return startIndexInColumn + (mastCount - 1) >= BOARD_SIZE;
037  }
038
039  private boolean checkForVertical() {
040    return !this.checkIfBreakingThroughBorders()
041        && !this.isShipOnShip()
042        && !this.isShipTouchOtherShip();
043  }
044
045  private boolean isOutOfRange() {
046    return shipOrientation.equals(ShipOrientation.VERTICAL)
047        ? shipStartIndex + mastCount > board.length
048        : shipStartIndex + (mastCount - 1) * BOARD_SIZE >= board.length;
049  }
050
051  private boolean isShipOnShip() {
052    return IntStream
053        .rangeClosed(shipStartIndex, shipStartIndex + mastCount - 1)
054        .anyMatch(p -> board[p].equals(FieldState.OCCUPIED));
055  }
056
057  private boolean isShipTouchOtherShip() {
058    return this.isShipAboveShip()
059        || this.isShipBelowShip()
060        || this.isShipOnTheRight()
061        || this.isShipOnTheLeft();
062  }
063
064  private boolean isShipAboveShip() {
065    int checkWidthInCell = 3;
066    return !this.isInLastRow()
067        && IntStream.iterate(shipStartIndex - BOARD_SIZE , i -> i + BOARD_SIZE)
068        .limit(checkWidthInCell)
069        .filter(i -> i > 0)
070        .filter(i -> i + mastCount < board.length)
071        .anyMatch(i -> board[i + mastCount].equals(FieldState.OCCUPIED));
072  }
073
074  private boolean isInLastRow() {
075    int lastRowModulo = 9;
076    return (shipStartIndex + (mastCount - 1)) % BOARD_SIZE == lastRowModulo;
077  }
078
079  private boolean isShipBelowShip() {
080    int checkWidthInCell = 3;
081    return !this.isInFirstRow() && shipStartIndex - 1 > 0
082        && IntStream.iterate(shipStartIndex - BOARD_SIZE , i -> i + BOARD_SIZE)
083        .limit(checkWidthInCell)
084        .filter(i -> i - 1 > 0)
085        .filter(i -> i < board.length)
086        .anyMatch(i -> board[i - 1].equals(FieldState.OCCUPIED));
087  }
088
089  private boolean isInFirstRow() {
090    return shipStartIndex % BOARD_SIZE == 0;
091  }
092
093  private boolean isShipOnTheRight() {
094    return IntStream.range(shipStartIndex, shipStartIndex + mastCount)
095        .filter(value -> value - BOARD_SIZE > 0)
096        .anyMatch(value -> board[value - BOARD_SIZE].equals(FieldState.OCCUPIED));
097  }
098
099  private boolean isShipOnTheLeft() {
100    return IntStream.range(shipStartIndex, shipStartIndex + mastCount)
101        .filter(value -> value > 0)
102        .filter(value -> value + BOARD_SIZE < board.length)
103        .anyMatch(value -> board[value + BOARD_SIZE].equals(FieldState.OCCUPIED));
104  }
105
106  private boolean checkForHorizontal() {
107    return !this.isShipOnShipHorizontal()
108        && !this.isShipTouchOtherShipHorizontal();
109  }
110
111  private boolean isShipTouchOtherShipHorizontal() {
112    return this.isShipBelowShipHorizontal()
113        || this.isShipAboveShipHorizontal()
114        || this.isShipOnTheRightHorizontal()
115        || this.isShipOnTheLeftHorizontal();
116  }
117
118  private boolean isShipOnShipHorizontal() {
119    return IntStream.iterate(shipStartIndex, n -> n + BOARD_SIZE)
120        .limit((mastCount - 1))
121        .anyMatch(n -> board[n].equals(FieldState.OCCUPIED));
122  }
123
124  private boolean isShipBelowShipHorizontal() {
125    return shipStartIndex % BOARD_SIZE != 9
126        && IntStream.iterate(shipStartIndex - BOARD_SIZE, n -> n + BOARD_SIZE)
127        .limit(mastCount + 2L)
128        .filter(n -> n > 0)
129        .filter(n -> n + 1 < board.length)
130        .anyMatch(n -> board[n + 1].equals(FieldState.OCCUPIED));
131  }
132
133  private boolean isShipAboveShipHorizontal() {
134    return !this.isInFirstRow()
135        && IntStream.iterate(shipStartIndex - BOARD_SIZE, n -> n + BOARD_SIZE)
136        .limit(mastCount + 2L)
137        .filter(n -> n > 0)
138        .filter(n -> n < board.length)
139        .anyMatch(n -> board[n - 1].equals(FieldState.OCCUPIED));
140  }
141
142  private boolean isShipOnTheRightHorizontal() {
143    return shipStartIndex + mastCount * BOARD_SIZE < board.length
144        && board[shipStartIndex + mastCount * BOARD_SIZE].equals(FieldState.OCCUPIED);
145  }
146
147  private boolean isShipOnTheLeftHorizontal() {
148    return shipStartIndex - BOARD_SIZE >= 1 && board[shipStartIndex - BOARD_SIZE].equals(FieldState.OCCUPIED);
149  }
150
151}
