001package pl.korotkevics.ships.client.client;
002
003import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
004
005import java.util.EnumMap;
006import java.util.Map;
007
008/**
009 * Helper class to build MessageHandler.
010 *
011 * @author Magdalena Aarsman
012 * @since 2018-01-02
013 */
014
015public class MessageHandlerBuilder {
016  private Map<Header, EventTrigger> triggers;
017
018  /**
019   * Set triggers map to be enum map with Header as key.
020   *
021   * @return MessageHandlerBuilder
022   */
023  public MessageHandlerBuilder withEnumMap() {
024    triggers = new EnumMap<>(Header.class);
025    return this;
026  }
027
028  /**
029   * Set default set of triggers and put it into map.
030   *
031   * @return MessageHandlerBuilder
032   */
033  public MessageHandlerBuilder withDefaultSetsOfTriggers() {
034    triggers.put(Header.OPPONENT_CONNECTED, new OpponentConnectedTrigger());
035    triggers.put(Header.SHOT, new OpponentShotTrigger());
036    triggers.put(Header.CONNECTION, new ConnectionEndTrigger());
037    triggers.put(Header.YOUR_TURN, new TurnTrigger());
038    triggers.put(Header.MISS, new MissShotTrigger());
039    triggers.put(Header.HIT, new HitShotTrigger());
040    triggers.put(Header.SHIP_DESTROYED, new ShipDestroyedTrigger());
041    triggers.put(Header.WIN, new WinTrigger());
042    triggers.put(Header.LOSE, new LoseTrigger());
043    triggers.put(Header.RANDOM_PLACEMENT, new RandomPlacementTrigger());
044    return this;
045  }
046
047  /**
048   * Build MessageHandler.
049   *
050   * @return MessageHandler
051   */
052  public MessageHandler build() {
053    return new MessageHandler(triggers);
054  }
055}
