001package pl.korotkevics.ships.client.client;
002
003import javafx.application.Platform;
004import javafx.scene.control.Button;
005import pl.korotkevics.ships.client.gui.events.RandomPlacementEvent;
006import pl.korotkevics.ships.shared.infra.communication.api.Message;
007
008/**
009 * Enable to fire event reacting to missing shot.
010 *
011 * @author Magdalena Aarsman
012 * @since 2017-01-15
013 */
014public class RandomPlacementTrigger implements EventTrigger {
015  @Override
016  public void fire(final Button button, final Message message) {
017    Platform.runLater(() ->
018        button.fireEvent(new RandomPlacementEvent(message.getFleet())));
019  }
020}
