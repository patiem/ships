080  private void initializeTurn(final boolean myTurn) {
081    if (!myTurn) {
082      opponentBoard.setDisable(true);
083      final double opacity = 0.4;
084      opponentBoard.setOpacity(opacity);
085    }
086  }
087
088  void initializeClient() {
089    getClient().setEventTrigger(eventButton);
090  }
091
092  void initializeBoards(GridPane yourBoard) {
093    final int margin = 50;
094    final NumberBinding allRectanglesWidth = Bindings.min(this.yourBoard.heightProperty(),
095        this.yourBoard.widthProperty().add(-margin));
096    final NumberBinding allRectanglesHeight = Bindings.min(this.yourBoard.heightProperty(),
097        this.yourBoard.widthProperty()).add(-margin);
098
099    for (int i = 0; i < BOARD_SIZE; i++) {
100      for (int j = 0; j < BOARD_SIZE; j++) {
101        final int shotIndex = j * BOARD_SIZE + i;
102
103        Rectangle yourRect = getYourRect(allRectanglesWidth, allRectanglesHeight);
104        Rectangle opponentRect = getOpponentRectangle(allRectanglesWidth,
105            allRectanglesHeight, shotIndex);
106
107        this.yourBoard.add(yourRect, i, j);
108        opponentBoard.add(opponentRect, i, j);
109
110        GridPane.setHalignment(yourRect, HPos.CENTER);
111        GridPane.setHalignment(opponentRect, HPos.CENTER);
112      }
113    }
114
115    copyYourBoard(yourBoard);
116  }
117
118  private void copyYourBoard(GridPane yourBoard) {
119    for (int i = 1; i < yourBoard.getChildren().size(); i++) {
120      if (((Rectangle) yourBoard.getChildren().get(i)).getFill() == Color.GREEN) {
121        ((Rectangle) this.yourBoard.getChildren().get(i)).setFill(Color.GREEN);
122      }
123    }
124  }
125
126  private Client getClient() {
127    MainController mainController = (MainController) mainAnchorPane.getParent().getUserData();
128    return mainController.getClient();
129  }
130
131  private Rectangle getYourRect(NumberBinding allRectanglesWidth,
132                                NumberBinding allRectanglesHeight) {
133    final int initialSize = 15;
134    Rectangle yourRect = new Rectangle(initialSize, initialSize, Color.GRAY);
135    yourRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
136    yourRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
137
138    return yourRect;
139  }
140
141  private Rectangle getOpponentRectangle(NumberBinding allRectanglesWidth,
142                                         NumberBinding allRectanglesHeight,
143                                         final int shotIndex) {
144    final int initialSize = 15;
145    Rectangle opponentRect = new Rectangle(initialSize, initialSize, Color.GRAY);
146
147    opponentRect.setOnMouseClicked((MouseEvent mouseEvent) -> {
148      opponentBoard.setDisable(true);
149      if (opponentRect.getFill().equals(Color.GRAY)) {
150        opponentRect.setFill(Color.BLACK);
151      }
152      this.shotIndex = shotIndex;
153      getClient().sendShot(shotIndex);
154    });
155
156    opponentRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
157    opponentRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
158
159    return opponentRect;
160  }
161
162  private void opponentWithdraw() {
163    getClient().closeClient();
164    loadWithdrawScreen();
165  }
166
167  private void loadWithdrawScreen() {
168    try {
169      final String opponentWithdrawUrl = "/fxml/opponentWithdraw.fxml";
170      final FXMLLoader opponentWithdrawLoader =
171          new FXMLLoader(getClass().getResource(opponentWithdrawUrl));
172      opponentWithdrawLoader.setResources(this.resourceBundle);
173      final Parent opponentWithdraw = opponentWithdrawLoader.load();
174      final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
175      final int sceneHeight = 400;
176      final int sceneWidth = 600;
177      final double margin = 0.0;
178
179      Stage stage = (Stage) mainPane.getScene().getWindow();
180      stage.setHeight(sceneHeight);
181      stage.setWidth(sceneWidth);
182
183      mainPane.getChildren().clear();
184      mainPane.getChildren().setAll(opponentWithdraw);
185
186      AnchorPane.setTopAnchor(opponentWithdraw, margin);
187      AnchorPane.setBottomAnchor(opponentWithdraw, margin);
188      AnchorPane.setLeftAnchor(opponentWithdraw, margin);
189      AnchorPane.setRightAnchor(opponentWithdraw, margin);
190
191    } catch (IOException e) {
192      logger.error(e.getMessage());
193    }
194  }
195
196  private void setOpponentShot(final int shotIndex) {
197    final int shotIndexInGrid = convertToGridIndex(shotIndex);
198    final Rectangle rec = (Rectangle) (yourBoard.getChildren().get(shotIndexInGrid));
199    if (rec.getFill() == Color.GREEN) {
200      rec.setFill(Color.RED);
201    } else if (rec.getFill().equals(Color.GRAY)) {
202      rec.setFill(Color.BLACK);
203    }
204  }
205
206  private int convertToGridIndex(final int shotIndex) {
207    final int column = shotIndex / BOARD_SIZE;
208    final int row = shotIndex - (column * BOARD_SIZE);
209    return row * BOARD_SIZE + column + 1;
210  }
211
212  private void changeTurn() {
213    final double opacity = 0.4;
214    this.opponentBoard.setDisable(true);
215    this.opponentBoard.setOpacity(opacity);
216    this.infoLabel.setText(this.resourceBundle.getString("youMiss")
217        + " "
218        + this.resourceBundle.getString("opponentAction"));
219  }
220
221  private void markAsHit() {
222    final int shotIndexInGrid = this.convertToGridIndex(shotIndex);
223    final Rectangle rec = (Rectangle) (opponentBoard.getChildren().get(shotIndexInGrid));
224    rec.setFill(Color.RED);
225    this.hitShot = true;
226  }
227
228  private void setMyTurn() {
229    final double noOpacity = 1.0;
230    opponentBoard.setDisable(false);
231    opponentBoard.setOpacity(noOpacity);
232    this.infoLabel.setText(this.resourceBundle.getString("yourTurn"));
233    if (this.hitShot) {
234      this.infoLabel.setText(this.infoLabel.getText()
235          + " "
236          + this.resourceBundle.getString("youHit"));
237      this.hitShot = false;
238    } else if (this.shipDestroyed) {
239      this.infoLabel.setText(this.infoLabel.getText()
240          + " "
241          + this.resourceBundle.getString("shipDestroyed"));
242      this.shipDestroyed = false;
243    }
244  }
245
246  @Override
247  public void initialize(final URL location, final ResourceBundle resources) {
248    this.resourceBundle = resources;
249    eventButton.addEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
250        opponentConnectedEvent -> opponentWithdraw());
251    eventButton.addEventHandler(OpponentShotEvent.OPPONENT_SHOT,
252        opponentShotEvent -> setOpponentShot(opponentShotEvent.getShotIndex()));
253    eventButton.addEventHandler(TurnChangeEvent.TURN_EVENT, event -> setMyTurn());
254    eventButton.addEventHandler(MissShotEvent.MISS_SHOT, event -> changeTurn());
255    eventButton.addEventHandler(HitShotEvent.HIT_SHOT, event -> markAsHit());
256    eventButton.addEventHandler(WinEvent.GAME_WIN, event -> renderAsWin());
257    eventButton.addEventHandler(LossEvent.GAME_LOSS, event -> renderAsLoss());
258    eventButton.addEventHandler(ShipDestroyedEvent.SHIP_DESTROYED, event -> this.shipDestroyed());
259    buttonEnd.setOnAction(actionEvent -> this.endTheGame());
260    infoLabel.setText(this.resourceBundle.getString("opponentAction"));
261    initializeTurn(false);
262  }
263
264  private void shipDestroyed() {
265    final int shotIndexInGrid = this.convertToGridIndex(shotIndex);
266    final Rectangle rec = (Rectangle) (opponentBoard.getChildren().get(shotIndexInGrid));
267    rec.setFill(Color.RED);
268    shipDestroyed = true;
269    SunkenShipMarker shipDestroyer =
270        new SunkenShipMarker(new GridToBoardConverter(opponentBoard).convert(), shotIndexInGrid - 1);
271    this.destroyShip(shipDestroyer.getIndexToColor());
272  }
273
274  private void endTheGame() {
275    Platform.exit();
276  }
277
278  private void renderAsWin() {
279    this.prepareToRenderAnyPossibleResult();
280    this.renderSpecificResult("youWin");
281    this.endStack.setVisible(true);
282    this.buttonEnd.setDisable(false);
283  }
284
285  private void renderAsLoss() {
286    this.prepareToRenderAnyPossibleResult();
287    this.renderSpecificResult("youLose");
288    this.endStack.setVisible(true);
289    this.buttonEnd.setDisable(false);
290  }
291
292  private void renderSpecificResult(String key) {
293    this.winLabel.setText(this.resourceBundle.getString(key));
294  }
295
296  private void prepareToRenderAnyPossibleResult() {
297    this.disableWithdrawalPossibility();
298    this.disableBoards();
299    this.infoLabel.setText("");
300  }
301
302  private void disableBoards() {
303    final double opacity = 0.4;
304    this.yourBoard.setDisable(true);
305    this.yourBoard.setOpacity(opacity);
306    this.opponentBoard.setDisable(true);
307    this.opponentBoard.setOpacity(opacity);
308    this.winLabel.setStyle("-fx-color: green");
309  }
310
311  private void disableWithdrawalPossibility() {
312    this.eventButton.removeEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
313        opponentConnectedEvent -> opponentWithdraw());
314  }
315
316  private void destroyShip(Set<Integer> indexes){
317    indexes.forEach(i -> ((Rectangle )this.opponentBoard.getChildren().get(i + 1)).setFill(Color.BLACK));
318  }
