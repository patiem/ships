001package pl.korotkevics.ships.client.gui.controllers;
002
003import javafx.fxml.FXML;
004import javafx.fxml.FXMLLoader;
005import javafx.fxml.Initializable;
006import javafx.scene.Parent;
007import javafx.scene.layout.AnchorPane;
008import javafx.scene.layout.Pane;
009import pl.korotkevics.ships.client.client.Client;
010import pl.korotkevics.ships.shared.infra.logging.api.Target;
011import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
012
013import java.io.IOException;
014import java.net.URL;
015import java.util.ResourceBundle;
016
017/**
018 * Server Not Responding window controller.
019 *
020 * @author Magdalena Aarsman
021 * @since 2017-12-17
022 */
023
024public class ServerNotRespondingController implements Initializable {
025  
026  private static final Target logger = new SharedLogger(Client.class);
027  
028  @FXML
029  private AnchorPane mainAnchorPane;
030  private ResourceBundle resourceBundle;
031  
032  @FXML
033  private void onTryAgainPressed() {
034    try {
035      final String connectWindowUrl = "/fxml/connectWindow.fxml";
036      final FXMLLoader connectLoader = new FXMLLoader(getClass().getResource(connectWindowUrl));
037      connectLoader.setResources(this.resourceBundle);
038      final Parent connectWindow = connectLoader.load();
039      final Pane mainPane = (Pane) mainAnchorPane.getParent();
040      mainPane.getChildren().clear();
041      mainPane.getChildren().setAll(connectWindow);
042    } catch (IOException e) {
043      logger.error(e.getMessage());
044    }
045  }
046  
047  @Override
048  public void initialize(final URL location, final ResourceBundle resources) {
049    this.resourceBundle = resources;
050  }
051}
