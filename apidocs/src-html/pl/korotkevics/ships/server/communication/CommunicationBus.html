001package pl.korotkevics.ships.server.communication;
002
003import pl.korotkevics.ships.shared.infra.communication.api.Message;
004import pl.korotkevics.ships.shared.infra.logging.api.Target;
005import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
006
007import java.io.IOException;
008import java.net.Socket;
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * Starts a server and gets its clients' sockets
014 * and adds them wrapped into own list.
015 *
016 * @author Piotr Czy≈º, Magdalena Aarsman, Sandor Korotkevics
017 * @since 2017-12-09
018 */
019public class CommunicationBus {
020
021
022  private final Target logger = new SharedLogger(CommunicationBus.class);
023  private final List<Socket> serverClients;
024  private List<WrappedClient> clients;
025
026  /**
027   * Create communication bus instance.
028   *
029   * @param serverClients List of clients sockets
030   */
031  public CommunicationBus(final List<Socket> serverClients) {
032    this.serverClients = serverClients;
033  }
034
035  /**
036   * It ask AppServer for clients and then wrap them into wrapped client.
037   */
038  public void start() {
039    clients = new ArrayList<>();
040    serverClients.forEach(this::wrapClient);
041    logger.info("Communication bus started..");
042  }
043
044  /**
045   * Close clients sockets and clear list of wrapped clients.
046   */
047  public void stop() {
048    clients.forEach(WrappedClient::close);
049    clients.clear();
050  }
051
052  private void wrapClient(final Socket socketClient) {
053    WrappedClient wrappedClient = new WrappedClient(socketClient);
054    clients.add(wrappedClient);
055  }
056
057  /**
058   * It receives message from a player.
059   *
060   * @param sender player who sent a message
061   * @return received message
062   */
063  public Message receive(final WrappedClient sender) {
064    return sender.receive();
065  }
066
067  /**
068   * It sends message to wrapped client.
069   *
070   * @param recipient message addressee
071   * @param response  message to send
072   */
073  public void send(final WrappedClient recipient, final Message response) {
074    recipient.send(response);
075  }
076
077  void sendToAll(final Message message) {
078    clients.forEach(wrappedClient -> send(wrappedClient, message));
079  }
080
081  /**
082   * Gets player which was connected earlier.
083   *
084   * @return WrappedClient
085   */
086  public WrappedClient getFirstClient() {
087    return this.clients.get(0);
088  }
089
090  /**
091   * Gets player which was connected as second one.
092   *
093   * @return WrappedClient
094   */
095  public WrappedClient getSecondClient() {
096    return this.clients.get(1);
097  }
098}
