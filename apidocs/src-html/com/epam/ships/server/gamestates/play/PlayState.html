001package com.epam.ships.server.gamestates.play;
002
003import com.epam.ships.fleet.Fleet;
004import com.epam.ships.infra.communication.api.Message;
005import com.epam.ships.infra.communication.api.message.Header;
006import com.epam.ships.infra.logging.api.Target;
007import com.epam.ships.infra.logging.core.SharedLogger;
008import com.epam.ships.server.CommunicationBus;
009import com.epam.ships.server.MessageSender;
010import com.epam.ships.server.TurnManager;
011import com.epam.ships.server.gamestates.GameEndWithWalkoverState;
012import com.epam.ships.server.gamestates.GameEndWithWinState;
013import com.epam.ships.server.gamestates.GameState;
014
015import java.util.List;
016
017/**
018 * Main class where the play goes.
019 *
020 * @author Piotr Czy≈º
021 * @since 2018-01-02
022 */
023public class PlayState implements GameState {
024  private final MessageReceiver messageReceiver;
025  private final ShotHandler shotHandler;
026  private final Target logger = new SharedLogger(PlayState.class);
027  private final TurnManager turnManager;
028  private CommunicationBus communicationBus;
029  private boolean isGameWon;
030
031  /**
032   * Crates PlayState instance.
033   *
034   * @param communicationBus client server communication bus
035   * @param fleets           list of clients fleets
036   */
037  public PlayState(final CommunicationBus communicationBus,final List<Fleet> fleets) {
038    this.communicationBus = communicationBus;
039    this.messageReceiver = new MessageReceiver(communicationBus);
040    isGameWon = false;
041    this.turnManager = new TurnManager(communicationBus.getFirstClient(),
042        communicationBus.getSecondClient());
043    this.shotHandler = new ShotHandler(communicationBus, turnManager, fleets);
044  }
045
046  /**
047   * Process gameplay until its have a winner.
048   *
049   * @return GameState, depends on game flow it can be GameEndWithWalkoverState
050   *         or GameEndWithWinState.
051   */
052  @Override
053  public GameState process() {
054    sendYourTurnMessage();
055    messageReceiver.receive(turnManager.getCurrentPlayer());
056    if (messageReceiver.isAShot()) {
057      Message shot = messageReceiver.getMessage();
058      isGameWon = shotHandler.handle(turnManager.isCurrentPlayerFirstPlayer(), shot);
059    } else {
060      return new GameEndWithWalkoverState(communicationBus);
061    }
062    if (isGameWon) {
063      return new GameEndWithWinState(communicationBus, turnManager);
064    }
065    rest();
066    return this;
067  }
068
069  private void sendYourTurnMessage() {
070    MessageSender messageSender = new MessageSender(communicationBus, logger);
071    messageSender.send(turnManager.getCurrentPlayer(), Header.YOUR_TURN);
072  }
073
074  private void rest() {
075    final long restTime = 300;
076    try {
077      Thread.sleep(restTime);
078    } catch (final InterruptedException e) {
079      logger.error(e.getMessage());
080      Thread.currentThread().interrupt();
081    }
082  }
083}
