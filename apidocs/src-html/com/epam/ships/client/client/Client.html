001package com.epam.ships.client.client;
002
003import com.epam.ships.shared.fleet.Fleet;
004import com.epam.ships.shared.infra.communication.api.Message;
005import com.epam.ships.shared.infra.communication.api.io.Receiver;
006import com.epam.ships.shared.infra.communication.api.io.Sender;
007import com.epam.ships.shared.infra.communication.api.message.Author;
008import com.epam.ships.shared.infra.communication.api.message.Header;
009import com.epam.ships.shared.infra.communication.core.json.io.JsonReceiver;
010import com.epam.ships.shared.infra.communication.core.json.io.JsonSender;
011import com.epam.ships.shared.infra.communication.core.message.MessageBuilder;
012import com.epam.ships.shared.infra.logging.api.Target;
013import com.epam.ships.shared.infra.logging.core.SharedLogger;
014import javafx.scene.control.Button;
015
016import java.io.IOException;
017import java.net.InetAddress;
018import java.net.InetSocketAddress;
019import java.net.Socket;
020
021/**
022 * Enable communication with server side.
023 *
024 * @author Magdalena Aarsman
025 * @since 2017-12-13
026 */
027public class Client implements Runnable {
028  private static final Target logger = new SharedLogger(Client.class);
029  private final MessageHandler messageHandler;
030  private Socket clientSocket;
031  private volatile boolean shouldRun;
032
033  /**
034   * Creates Client instance.
035   * @param messageHandler - messageHandler instance
036   * @param shouldRun - flag informing if client should start/run
037   */
038  public Client(final MessageHandler messageHandler, boolean shouldRun) {
039    this.messageHandler = messageHandler;
040    this.shouldRun = shouldRun;
041  }
042
043  /**
044   * Connecting client with server using providing address and port.
045   *
046   * @param ipAddress - server ip address
047   * @param port      - server port
048   * @param socket    - socket to connect on.
049   * @return true if success, false on failure
050   */
051  public boolean connect(final String ipAddress, final int port, Socket socket) {
052    try {
053      clientSocket = socket;
054      final InetAddress address = InetAddress.getByName(ipAddress);
055      final int connectionTimeout = 500;
056      clientSocket.connect(new InetSocketAddress(address, port), connectionTimeout);
057    } catch (IOException | IllegalArgumentException e) {
058      logger.error(e.getMessage());
059      return false;
060    }
061    return true;
062  }
063
064  /**
065   * Override run method of Runnable interface.
066   * Setting name of thread to "Client listen Thread" and start listen loop.
067   */
068  @Override
069  public void run() {
070    Thread.currentThread().setName("Client listen Thread");
071    listenLoop();
072  }
073
074  /**
075   * Closing client, stopping listen thread.
076   */
077  public void closeClient() {
078    this.shouldRun = false;
079    if (clientSocket == null) {
080      return;
081    }
082
083    try {
084      clientSocket.close();
085    } catch (IOException e) {
086      logger.error(e.getMessage());
087    }
088  }
089
090  private void listenLoop() {
091
092    while (shouldRun && !messageHandler.isEndConnectionTriggered()) {
093      try {
094        Receiver receiver = new JsonReceiver(clientSocket.getInputStream());
095        Message message = receiver.receive();
096
097        logger.info(message);
098
099        messageHandler.handle(message);
100
101      } catch (IOException | IllegalStateException e) {
102        logger.error(e.getMessage());
103      }
104
105      rest();
106    }
107  }
108
109  private void rest() {
110    try {
111      Thread.sleep(300);
112    } catch (InterruptedException e) {
113      logger.info(e.getMessage());
114      Thread.currentThread().interrupt();
115    }
116  }
117
118  /**
119   * Sending message containing shot index to server.
120   *
121   * @param shotIndex - index of last shot
122   */
123  public void sendShot(int shotIndex) {
124    try {
125      Sender sender = new JsonSender(clientSocket.getOutputStream());
126      Message shot = new MessageBuilder().withHeader(Header.SHOT)
127          .withAuthor(Author.CLIENT)
128          .withStatement(String.valueOf(shotIndex))
129          .build();
130      sender.send(shot);
131    } catch (IOException e) {
132      logger.error(e.getMessage());
133    }
134  }
135
136  /**
137   * Sending message containing fleet placement to server.
138   *
139   * @param fleet - Fleet object representing user fleet.
140   */
141  public void sendFleet(Fleet fleet) {
142    try {
143      Sender sender = new JsonSender(clientSocket.getOutputStream());
144      Message fleetMsg = new MessageBuilder()
145          .withHeader(Header.PLACEMENT)
146          .withAuthor(Author.CLIENT)
147          .withFleet(fleet)
148          .build();
149      sender.send(fleetMsg);
150    } catch (IOException e) {
151      logger.error(e.getMessage());
152    }
153  }
154
155  /**
156   * Set handle to gui to trigger some events in response for server message.
157   *
158   * @param button - gui handler as button.
159   */
160  public void setEventTrigger(Button button) {
161    messageHandler.setCurrentEventButton(button);
162  }
163}
