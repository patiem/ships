3   import com.google.gson.Gson;
4   import com.google.gson.GsonBuilder;
5   import com.google.gson.JsonElement;
6   
7   import pl.korotkevics.ships.shared.infra.communication.api.Message;
8   import pl.korotkevics.ships.shared.infra.communication.api.conversion.Encoder;
9   import pl.korotkevics.ships.shared.infra.communication.core.message.BaseMessage;
10  
11  /**
12   * Converts a BaseMessage instance into a JsonElement.
13   * @author Sandor Korotkevics
14   * @see Encoder
15   * @see BaseMessage
16   * @since 2017-12-10
17   */
18  public class JsonEncoder implements Encoder<JsonElement> {
19  
20    /**
21     * Converts an instance of a class implementing
22     * BaseMessage interface into a JsonElement instance.
23     *
24     * @param message an instance of a class implementing
25     *                BaseMessage interface.
26     * @return a result of conversion of an instance of a class implementing
27     *     BaseMessage interface into a JsonElement instance.
28     */
29    @Override
30    public JsonElement encode(Message message) {
31      Gson gson = new GsonBuilder()
32          .enableComplexMapKeySerialization()
33          .create();
34      return gson.toJsonTree(message, BaseMessage.class);
35    }
36  
37  }
