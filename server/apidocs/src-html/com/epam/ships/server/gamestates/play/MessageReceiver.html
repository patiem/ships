001package com.epam.ships.server.gamestates.play;
002
003import com.epam.ships.infra.communication.api.Message;
004import com.epam.ships.infra.communication.api.message.Header;
005import com.epam.ships.infra.logging.api.Target;
006import com.epam.ships.infra.logging.core.SharedLogger;
007import com.epam.ships.server.CommunicationBus;
008import com.epam.ships.server.WrappedClient;
009
010/**
011 * Receive message from client.
012 *
013 * @author Piotr Czy≈º
014 * @since 2018-01-02
015 */
016public class MessageReceiver {
017  private final Target logger = new SharedLogger(MessageReceiver.class);
018  private CommunicationBus communicationBus;
019  private Message lastMessage;
020
021  MessageReceiver(final CommunicationBus communicationBus) {
022    this.communicationBus = communicationBus;
023  }
024
025  void receive(final WrappedClient player) {
026    lastMessage = communicationBus.receive(player);
027    logger.info(lastMessage);
028  }
029
030  boolean isAShot() {
031    return Header.SHOT.equals(lastMessage.getHeader());
032  }
033
034  Message getMessage() {
035    return lastMessage;
036  }
037}
