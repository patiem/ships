001package com.epam.ships.server.gamestates;
002
003import com.epam.ships.infra.logging.api.Target;
004import com.epam.ships.infra.logging.core.SharedLogger;
005import com.epam.ships.server.CommunicationBus;
006
007/**
008 * Represents game state where one of players quit in the middle of game.
009 *
010 * @author Piotr Czy≈º
011 * @since 2018-01-02
012 */
013public class GameEndWithWalkoverState implements GameState {
014  private final Target logger = new SharedLogger(GameEndWithWalkoverState.class);
015  private final CommunicationBus communicationBus;
016
017  public GameEndWithWalkoverState(final CommunicationBus communicationBus) {
018    this.communicationBus = communicationBus;
019  }
020
021  /**
022   * Process game end with walk-over.
023   *
024   * @return GameState - returns it self because there is no more game states.
025   */
026  @Override
027  public GameState process() {
028    logger.info("Game ends - walk-over");
029    communicationBus.stop();
030    return this;
031  }
032
033  /**
034   * Set that game should not by continued since there is no opponent.
035   *
036   * @return false because that was
037   */
038  @Override
039  public boolean shouldBeContinued() {
040    return false;
041  }
042}
