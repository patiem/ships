001package pl.korotkevics.ships.server.communication;
002
003import pl.korotkevics.ships.shared.infra.communication.api.Message;
004import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
005import pl.korotkevics.ships.shared.infra.logging.api.Target;
006
007/**
008 * Sends messages to clients.
009 *
010 * @author Piotr Czy≈º
011 * @since 2018-01-02
012 */
013public class MessageSender {
014  private final MessageRepository messages;
015  private CommunicationBus communicationBus;
016  private Target logger;
017
018  /**
019   * Creates Message sender instance.
020   *
021   * @param communicationBus client server communication bus
022   * @param logger           logger comes from class witch want to sand proper message
023   */
024  public MessageSender(final CommunicationBus communicationBus, final Target logger) {
025    this.communicationBus = communicationBus;
026    this.logger = logger;
027    messages = new MessageRepository();
028  }
029
030  /**
031   * Get message by header from MessageRepository and send it to addressee.
032   *
033   * @param addressee message addressee
034   * @param header    header of message to send
035   */
036  public void send(final WrappedClient addressee, final Header header) {
037    communicationBus.send(addressee, messages.getMessage(header));
038    logger.info("Send message with header: " + header);
039  }
040
041  /**
042   * Send message to an addressee.
043   *
044   * @param addressee message addressee
045   * @param message   message to send
046   */
047  public void send(final WrappedClient addressee, final Message message) {
048    communicationBus.send(addressee, message);
049    logger.info("Send message with header: " + message.getHeader());
050  }
051
052  /**
053   * Get message by header from MessageRepository and send it to all connected clients.
054   *
055   * @param header header of message to send
056   */
057  public void sendToAll(final Header header) {
058    communicationBus.sendToAll(messages.getMessage(header));
059    logger.info("Send message to both players " + header);
060  }
061}
