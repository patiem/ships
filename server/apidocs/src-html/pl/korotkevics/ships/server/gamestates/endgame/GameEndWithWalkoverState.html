001package pl.korotkevics.ships.server.gamestates.endgame;
002
003import pl.korotkevics.ships.server.CommunicationBus;
004import pl.korotkevics.ships.server.gamestates.GameState;
005import pl.korotkevics.ships.shared.infra.logging.api.Target;
006import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
007
008/**
009 * Represents game state in which one of players quits in the middle of game.
010 *
011 * @author Piotr Czy≈º
012 * @since 2018-01-02
013 */
014public class GameEndWithWalkoverState implements GameState {
015  private final Target logger = new SharedLogger(GameEndWithWalkoverState.class);
016  private final CommunicationBus communicationBus;
017
018  /**
019   * It creates state in which there is a walk-over.
020   *
021   * @param communicationBus client server communication bus
022   */
023  public GameEndWithWalkoverState(final CommunicationBus communicationBus) {
024    this.communicationBus = communicationBus;
025  }
026
027  /**
028   * Process game end with walk-over.
029   *
030   * @return return a state in which game will be discontinued
031   * @see #shouldBeContinued()
032   */
033  @Override
034  public GameState process() {
035    logger.info("Game ends - walk-over");
036    communicationBus.stop();
037    return this;
038  }
039
040  /**
041   * Set that game should not by continued since there is no opponent.
042   *
043   * @return false when game can not be continued because there is no opponent.
044   */
045  @Override
046  public boolean shouldBeContinued() {
047    return false;
048  }
049}
