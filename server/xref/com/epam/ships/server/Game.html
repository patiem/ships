1   package com.epam.ships.server;

3   import com.epam.ships.fleet.Damage;
4   import com.epam.ships.fleet.Fleet;
5   import com.epam.ships.fleet.Mast;
6   import com.epam.ships.infra.communication.api.Message;
7   import com.epam.ships.infra.communication.api.message.Author;
8   import com.epam.ships.infra.communication.api.message.Header;
9   import com.epam.ships.infra.communication.api.message.Status;
10  import com.epam.ships.infra.communication.core.message.MessageBuilder;
11  import com.epam.ships.infra.logging.api.Target;
12  import com.epam.ships.infra.logging.core.SharedLogger;

14  /**
15   * * It starts a communication bus
16   * and receives messages from it.
17   * @author Piotr Czy≈º
18   * @since 2017-12-09
19   */
20  class Game {
21  
22    private final CommunicationBus communicationBus;
23    private final Target logger = new SharedLogger(Game.class);
24    private final TurnManager turnManager;
25    private Fleet firstPlayerFleet;
26    private Fleet secondPlayerFleet;
27  
28    Game(CommunicationBus communicationBus) {
29      this.communicationBus = communicationBus;
30      this.turnManager = new TurnManager(communicationBus.getFirstClient(),
31          communicationBus.getSecondClient());
32    }
33  
34    void play() {
35      this.notifyPlayersThatTheyCanStartGame();
36      this.askPlayersForPlaceFleet();
37      this.receiveFleetFromBothPlayers();
38      this.rest();
39      this.sendYourTurnMessage();
40      boolean isClientConnected = true;
41      boolean isGameFinished = false;
42      while (!isGameFinished && isClientConnected) {
43        Message receivedShot = this.receiveShot();
44        isClientConnected = this.isClientConnected(receivedShot);
45        if (isClientConnected) {
46          if (this.turnManager.isCurrentPlayerFirstPlayer()) {
47            isGameFinished = this.handleShot(receivedShot, secondPlayerFleet);
48          } else {
49            isGameFinished = this.handleShot(receivedShot, firstPlayerFleet);
50          }
51        }
52        if (isGameFinished) {
53          this.handleEndOfGame();
54        }
55        this.rest();
56      }
57    }
58  
59    private boolean handleShot(Message receivedShot, Fleet fleet) {
60      final Mast mast = Mast.ofIndex(receivedShot.getStatement());
61      logger.debug(receivedShot.getStatement());
62      Damage damage = fleet.handleDamage(mast);
63      if (damage.equals(Damage.MISSED)) {
64        logger.debug("got here: missed");
65        this.handleMiss(receivedShot);
66      } else if (damage.equals(Damage.HIT)) {
67        logger.debug("got here: hit");
68        this.handleHit(receivedShot);
69      } else if (damage.equals(Damage.DESTRUCTED)) {
70        logger.debug("got here: destructed");
71        this.handleDamageShot(receivedShot);
72      }
73      return fleet.isDefeated();
74    }
75  
76    private void handleDamageShot(Message receivedShot) {
77      this.sendShipDestructedMessage();
78      this.sendShotToOpponent(receivedShot);
79    }
80  
81    private void handleEndOfGame() {
82      this.sendWonMessage();
83      this.sendLostMessage();
84    }
85  
86    private void sendWonMessage() {
87      final Message won = new MessageBuilder()
88          .withAuthor(Author.SERVER)
89          .withHeader(Header.WIN)
90          .build();
91      this.communicationBus.send(turnManager.getCurrentPlayer(), won);
92    }
93  
94    private void sendLostMessage() {
95      final Message lost = new MessageBuilder()
96          .withAuthor(Author.SERVER)
97          .withHeader(Header.LOSE)
98          .build();
99      this.communicationBus.send(turnManager.getOtherPlayer(), lost);
100   }
101 
102   private void sendShipDestructedMessage() {
103     final Message hit = new MessageBuilder()
104         .withAuthor(Author.SERVER)
105         .withHeader(Header.SHIP_DESTRUCTED)
106         .build();
107     this.communicationBus.send(turnManager.getCurrentPlayer(), hit);
108   }
109 
110   private void handleHit(Message receivedShot) {
111     this.sendHitMessage();
112     this.sendShotToOpponent(receivedShot);
113   }
114 
115   private void sendHitMessage() {
116     final Message hit = new MessageBuilder()
117         .withAuthor(Author.SERVER)
118         .withHeader(Header.HIT)
119         .build();
120     this.communicationBus.send(turnManager.getCurrentPlayer(), hit);
121   }
122 
123   private void handleMiss(Message receivedShot) {
124     this.sendMissMessage();
125     this.sendShotToOpponent(receivedShot);
126     this.turnManager.switchPlayer();
127     rest();
128     this.sendYourTurnMessage();
129   }
130 
131   private void sendMissMessage() {
132     final Message miss = new MessageBuilder()
133         .withAuthor(Author.SERVER)
134         .withHeader(Header.MISS)
135         .build();
136     this.communicationBus.send(turnManager.getCurrentPlayer(), miss);
137   }
138 
139   private void receiveFleetFromBothPlayers() {
140     this.firstPlayerFleet = this.receiveFloat();
141     this.turnManager.switchPlayer();
142     this.secondPlayerFleet = this.receiveFloat();
143     this.turnManager.switchPlayer();
144   }
145 
146   private Fleet receiveFloat() {
147     final Message fleet = this.communicationBus.receive(this.turnManager.getCurrentPlayer());
148     logger.info("Fleet received: " + fleet);
149     return fleet.getFleet();
150   }
151 
152   private void askPlayersForPlaceFleet() {
153     this.askForPlaceFleet();
154     this.turnManager.switchPlayer();
155     this.askForPlaceFleet();
156     this.turnManager.switchPlayer();
157   }
158 
159   private void askForPlaceFleet() {
160     final Message message = new MessageBuilder()
161         .withAuthor(Author.SERVER)
162         .withHeader(Header.PLACEMENT)
163         .build();
164     this.communicationBus.send(turnManager.getCurrentPlayer(), message);
165   }
166 
167   private void sendShotToOpponent(Message receivedShot) {
168     this.communicationBus.send(this.turnManager.getOtherPlayer(), receivedShot);
169     logger.info("Shot send");
170   }
171 
172   private Message receiveShot() {
173     final Message shot = this.communicationBus.receive(this.turnManager.getCurrentPlayer());
174     logger.info(shot);
175     return shot;
176   }
177 
178   private void sendYourTurnMessage() {
179     final Message turn = new MessageBuilder()
180         .withAuthor(Author.SERVER)
181         .withHeader(Header.YOUR_TURN)
182         .build();
183     this.communicationBus.send(turnManager.getCurrentPlayer(), turn);
184     logger.info("send your turn");
185   }
186 
187   private boolean isClientConnected(final Message messageReceived) {
188     boolean isClientConnected = true;
189     if (Header.CONNECTION.equals(messageReceived.getHeader())
190         && Status.END.equals(messageReceived.getStatus())) {
191       isClientConnected = false;
192     }
193     return isClientConnected;
194   }
195 
196   private void notifyPlayersThatTheyCanStartGame() {
197     this.opponentConnected();
198     this.turnManager.switchPlayer();
199     this.opponentConnected();
200     this.turnManager.switchPlayer();
201   }
202 
203   private void opponentConnected() {
204     final Message message = new MessageBuilder()
205         .withAuthor(Author.SERVER)
206         .withHeader(Header.OPPONENT_CONNECTED)
207         .withStatement("true")
208         .build();
209     this.communicationBus.send(this.turnManager.getCurrentPlayer(), message);
210   }
211 
212   private void rest() {
213     final long restTime = 300;
214     try {
215       Thread.sleep(restTime);
216     } catch (final InterruptedException e) {
217       logger.error(e.getMessage());
218       Thread.currentThread().interrupt();
219     }
220   }
221 }
