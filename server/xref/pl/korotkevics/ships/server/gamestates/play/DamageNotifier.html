8   /**
9    * Notification about strength of shot.
10   *
11   * @author Piotr Czy≈º
12   * @since 2018-01-02
13   */
14  abstract class DamageNotifier {
15    protected MessageSender messageSender;
16    protected TurnManager turnManager;
17    protected Header header;
18  
19    DamageNotifier(final MessageSender messageSender, final TurnManager turnManager) {
20      this.messageSender = messageSender;
21      this.turnManager = turnManager;
22    }
23  
24    protected void notify(final Message receivedShot) {
25      messageSender.send(turnManager.getCurrentPlayer(), header);
26      informOpponentAboutShot(receivedShot);
27    }
28  
29    void informOpponentAboutShot(final Message receivedShot) {
30      messageSender.send(turnManager.getOtherPlayer(), receivedShot);
31    }
32  }
