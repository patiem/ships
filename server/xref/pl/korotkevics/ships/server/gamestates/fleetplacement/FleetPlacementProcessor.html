15  /**
16   * Bases on header it returns generated fleet or received one.
17   *
18   * @author Piotr Czy≈º
19   * @since 2018-01-10
20   */
21  class FleetPlacementProcessor {
22  
23    private Target logger = new SharedLogger(FleetPlacementProcessor.class);
24    private final CommunicationBus communicationBus;
25    private MessageReceiver messageReceiver;
26  
27    FleetPlacementProcessor(final CommunicationBus communicationBus) {
28      this.communicationBus = communicationBus;
29      this.messageReceiver = new MessageReceiver(communicationBus);
30    }
31  
32    Fleet placeFleet(final WrappedClient wrappedClient) {
33      this.messageReceiver.receive(wrappedClient);
34      if (this.messageReceiver.isRandomPlacement()) {
35        FleetGenerator generator = new FleetGenerator();
36        Fleet fleet = generator.generateFleet();
37        this.send(wrappedClient, fleet);
38        logger.info(fleet.toString());
39        return fleet;
40      }
41      return receiveFloat();
42    }
43  
44    private void send(final WrappedClient wrappedClient, final Fleet fleet) {
45      MessageSender messageSender = new MessageSender(communicationBus, logger);
46      final Message message = new MessageBuilder()
47          .withAuthor(Author.SERVER)
48          .withHeader(Header.RANDOM_PLACEMENT)
49          .withFleet(fleet)
50          .build();
51      messageSender.send(wrappedClient, message);
52    }
53  
54    private Fleet receiveFloat() {
55      return this.messageReceiver.getMessage().getFleet();
56    }
57  }
