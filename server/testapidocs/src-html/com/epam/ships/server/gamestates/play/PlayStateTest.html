001package com.epam.ships.server.gamestates.play;
002
003import com.epam.ships.shared.fleet.Fleet;
004import com.epam.ships.shared.fleet.Mast;
005import com.epam.ships.shared.fleet.Ship;
006import com.epam.ships.shared.infra.communication.api.Message;
007import com.epam.ships.shared.infra.communication.api.message.Author;
008import com.epam.ships.shared.infra.communication.api.message.Header;
009import com.epam.ships.shared.infra.communication.core.message.MessageBuilder;
010import com.epam.ships.server.CommunicationBus;
011import com.epam.ships.server.WrappedClient;
012import com.epam.ships.server.gamestates.GameEndWithWalkoverState;
013import com.epam.ships.server.gamestates.GameEndWithWinState;
014import com.epam.ships.server.gamestates.GameState;
015import org.mockito.Mock;
016import org.testng.annotations.BeforeMethod;
017import org.testng.annotations.Test;
018
019import java.util.Arrays;
020import java.util.List;
021
022import static org.mockito.Mockito.mock;
023import static org.mockito.Mockito.when;
024import static org.mockito.MockitoAnnotations.initMocks;
025import static org.testng.Assert.assertEquals;
026
027@Test
028public class PlayStateTest {
029
030  @Mock
031  private CommunicationBus communicationBus;
032  @Mock
033  private Fleet fleet;
034  @Mock
035  private WrappedClient sender;
036  @Mock
037  private WrappedClient wrappedClient;
038
039  @BeforeMethod
040  void before(){
041    initMocks(this);
042  }
043
044
045  public void shouldKeepStateWhenNoWinner() {
046    Fleet secondFleet = Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("4"), Mast.ofIndex("5"))));
047    List<Fleet> fleetList = Arrays.asList(fleet, secondFleet);
048    processShot();
049    PlayState playState = new PlayState(communicationBus, fleetList);
050    //when
051    GameState gameState = playState.process();
052    //then
053    assertEquals(PlayState.class, gameState.getClass());
054  }
055
056  public void shouldProcessGameToWinState() {
057    //given
058    Fleet secondFleet = Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("4"))));
059    List<Fleet> fleetList = Arrays.asList(fleet, secondFleet);
060    processShot();
061    PlayState playState = new PlayState(communicationBus, fleetList);
062    //when
063    GameState gameState = playState.process();
064    //then
065    assertEquals(GameEndWithWinState.class, gameState.getClass());
066  }
067
068  public void shouldProcessGameToWalkover() {
069    Fleet secondFleet = mock(Fleet.class);
070    List<Fleet> fleetList = Arrays.asList(fleet, secondFleet);
071    Message shot = new MessageBuilder()
072        .withAuthor(Author.AUTO)
073        .withHeader(Header.CONNECTION)
074        .withStatement("END")
075        .build();
076    when(communicationBus.receive(sender)).thenReturn(shot);
077    when(communicationBus.getFirstClient()).thenReturn(sender);
078    when(communicationBus.getSecondClient()).thenReturn(wrappedClient);
079    PlayState playState = new PlayState(communicationBus, fleetList);
080    //when
081    GameState gameState = playState.process();
082    //then
083    assertEquals(GameEndWithWalkoverState.class, gameState.getClass());
084  }
085
086  private void processShot() {
087    when(communicationBus.receive(sender)).thenReturn(produceMessageWithShot());
088    when(communicationBus.getFirstClient()).thenReturn(sender);
089    when(communicationBus.getSecondClient()).thenReturn(wrappedClient);
090  }
091
092  private Message produceMessageWithShot() {
093    return new MessageBuilder()
094        .withAuthor(Author.CLIENT)
095        .withHeader(Header.SHOT)
096        .withStatement("4")
097        .build();
098  }
099}
