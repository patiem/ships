001package com.epam.ships.server;
002
003import com.epam.ships.server.gamestates.GameState;
004import org.testng.annotations.Test;
005
006import static org.mockito.Mockito.*;
007
008@Test
009public class GameTest {
010
011  public void shouldEndWhenShouldBeContinuedConditionIsFalse() {
012    //given
013    GameState initState = mock(GameState.class);
014    GameState secondState = mock(GameState.class);
015
016    when(initState.process()).thenReturn(secondState);
017    when(secondState.process()).thenReturn(secondState);
018
019    when(initState.shouldBeContinued()).thenReturn(true);
020    when(secondState.shouldBeContinued()).thenReturn(false);
021
022    //when
023    new Game(initState).loop();
024
025    //then
026    verify(initState, times(1)).process();
027    verify(secondState, times(1)).process();
028  }
029
030}
