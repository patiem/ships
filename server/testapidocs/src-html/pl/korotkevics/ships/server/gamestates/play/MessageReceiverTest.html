001package pl.korotkevics.ships.server.gamestates.play;
002
003import pl.korotkevics.ships.shared.infra.communication.api.Message;
004import pl.korotkevics.ships.shared.infra.communication.api.message.Author;
005import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
006import pl.korotkevics.ships.shared.infra.communication.core.message.MessageBuilder;
007import pl.korotkevics.ships.server.CommunicationBus;
008import pl.korotkevics.ships.server.WrappedClient;
009import org.testng.annotations.Test;
010
011import static org.mockito.Mockito.*;
012import static org.testng.Assert.assertFalse;
013import static org.testng.Assert.assertTrue;
014
015@Test
016public class MessageReceiverTest {
017  public void shouldReceiveMessageFromPlayer() {
018    //given
019    CommunicationBus communicationBus = mock(CommunicationBus.class);
020    WrappedClient wrappedClient = mock(WrappedClient.class);
021
022    //when
023    MessageReceiver messageReceiver = new MessageReceiver(communicationBus);
024    messageReceiver.receive(wrappedClient);
025
026    //then
027    verify(communicationBus, times(1)).receive(wrappedClient);
028  }
029
030  public void shouldBeAShotIfLastMessageWasShot() {
031    //given
032    Message lastMessage = new MessageBuilder()
033        .withAuthor(Author.SERVER)
034        .withHeader(Header.SHOT)
035        .build();
036    CommunicationBus communicationBus = mock(CommunicationBus.class);
037    WrappedClient wrappedClient = mock(WrappedClient.class);
038    when(communicationBus.receive(wrappedClient)).thenReturn(lastMessage);
039
040    MessageReceiver messageReceiver = new MessageReceiver(communicationBus);
041    messageReceiver.receive(wrappedClient);
042
043    //when
044    boolean result = messageReceiver.isAShot();
045
046    //then
047    assertTrue(result);
048  }
049
050  public void shouldNotBeAShotIfLastMessageWasNotShot() {
051    //given
052    Message lastMessage = new MessageBuilder()
053        .withAuthor(Author.AUTO)
054        .withHeader(Header.CONNECTION)
055        .build();
056    CommunicationBus communicationBus = mock(CommunicationBus.class);
057    WrappedClient wrappedClient = mock(WrappedClient.class);
058    when(communicationBus.receive(wrappedClient)).thenReturn(lastMessage);
059
060    MessageReceiver messageReceiver = new MessageReceiver(communicationBus);
061    messageReceiver.receive(wrappedClient);
062    //when
063    boolean result = messageReceiver.isAShot();
064
065    //then
066    assertFalse(result);
067  }
068}
