001package com.epam.ships.client.gui.controllers;
002
003import com.epam.ships.client.client.Client;
004import com.epam.ships.client.gui.events.OpponentConnectedEvent;
005import com.epam.ships.client.validators.PortValidator;
006import com.epam.ships.infra.logging.api.Target;
007import com.epam.ships.infra.logging.core.SharedLogger;
008import javafx.fxml.FXML;
009import javafx.fxml.FXMLLoader;
010import javafx.geometry.Rectangle2D;
011import javafx.scene.Parent;
012import javafx.scene.control.Button;
013import javafx.scene.control.Label;
014import javafx.scene.control.TextField;
015import javafx.scene.image.ImageView;
016import javafx.scene.layout.AnchorPane;
017import javafx.scene.layout.GridPane;
018import javafx.scene.layout.Pane;
019import javafx.scene.layout.VBox;
020import javafx.stage.Screen;
021import javafx.stage.Stage;
022
023import java.io.IOException;
024
025/**
026 * Connecting window controller.
027 * @author Magdalena Aarsman
028 * @since 2017-12-14
029 */
030public class StartWindowController {
031
032  private static final Target logger = new SharedLogger(Client.class);
033
034  private Client client;
035
036  @FXML
037  private GridPane gridPane;
038
039  @FXML
040  private ImageView imCannon;
041
042  @FXML
043  private VBox vbWheel;
044
045  @FXML
046  private TextField textFieldServerAddress;
047
048  @FXML
049  private TextField textFieldServerPort;
050
051  @FXML
052  private AnchorPane mainAnchorPane;
053
054  @FXML
055  private Label labelInvalidPort;
056
057  @FXML
058  private Button eventButton;
059
060  private PortValidator portValidator;
061
062  @FXML
063  void initialize() {
064    textFieldServerPort.textProperty().addListener((observableValue, s1, t1) -> {
065      if (!labelInvalidPort.getText().isEmpty()) {
066        labelInvalidPort.setText("");
067      }
068    });
069
070    portValidator = new PortValidator();
071    eventButton.addEventHandler(OpponentConnectedEvent.OPPONENT_CONNECTED,
072        event -> loadFleetPlacementWindow());
073
074    final String defaultHost = "127.0.0.1";
075    final String defaultPort = "8189";
076
077    textFieldServerAddress.setText(defaultHost);
078    textFieldServerPort.setText(defaultPort);
079  }
080
081  private void initializeClient() {
082    MainController mainController = (MainController) mainAnchorPane.getParent().getUserData();
083    this.client = mainController.getClient();
084    if (client == null) {
085      throw new IllegalStateException("client is not initialized!");
086    }
087
088    client.setEventTrigger(eventButton);
089  }
090
091  @FXML
092  private void onConnectPressed() {
093    try {
094      initializeClient();
095    } catch (IllegalStateException e) {
096      logger.error(e.getMessage());
097      return;
098    }
099
100    final String serverAddress = textFieldServerAddress.getText();
101    logger.info("server address: " + serverAddress);
102
103    int port;
104    try {
105      port = portValidator.asInt(textFieldServerPort.getText());
106    } catch (IllegalArgumentException e) {
107      labelInvalidPort.setText("invalid port number");
108      return;
109    }
110    logger.info("server port: " + port);
111
112    showLoadingWheel();
113    final boolean isConnected = client.connect(serverAddress, port);
114
115    if (!isConnected) {
116      loadServerNotResponseView();
117    } else {
118      Thread clientThread = new Thread(client);
119      clientThread.start();
120    }
121  }
122
123  private void showLoadingWheel() {
124    final double opacity = 0.4;
125
126    gridPane.setDisable(true);
127    gridPane.setOpacity(opacity);
128    imCannon.setOpacity(opacity);
129    vbWheel.setVisible(true);
130  }
131
132  private void loadServerNotResponseView() {
133    try {
134      final String serverNotRespondingUrl = "/fxml/serverNotResponding.fxml";
135      final FXMLLoader notResponseLoader =
136          new FXMLLoader(getClass().getResource(serverNotRespondingUrl));
137      final Parent notResponse = notResponseLoader.load();
138      final Pane mainPane = (Pane) mainAnchorPane.getParent();
139      mainPane.getChildren().clear();
140      mainPane.getChildren().setAll(notResponse);
141    } catch (IOException e) {
142      logger.error(e.getMessage());
143    }
144  }
145
146  private void loadFleetPlacementWindow() {
147    try {
148      final String gameWindowUrl = "/fxml/fleetPlacement.fxml";
149      final FXMLLoader gameWindowLoader = new FXMLLoader(getClass().getResource(gameWindowUrl));
150      final Parent gameWindow = gameWindowLoader.load();
151      final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
152
153      Rectangle2D screenBounds = Screen.getPrimary().getVisualBounds();
154
155      Stage stage = (Stage) mainPane.getScene().getWindow();
156      final int shrinkSize = 300;
157      stage.setMinHeight(screenBounds.getHeight() - shrinkSize);
158      stage.setMinWidth(screenBounds.getWidth() - shrinkSize);
159
160      mainPane.getChildren().clear();
161      mainPane.getChildren().setAll(gameWindow);
162
163      FleetPlacementController fleetPlacementController = gameWindowLoader.getController();
164      fleetPlacementController.initializeClient();
165
166      final double margin = 0.0;
167      AnchorPane.setTopAnchor(gameWindow, margin);
168      AnchorPane.setBottomAnchor(gameWindow, margin);
169      AnchorPane.setLeftAnchor(gameWindow, margin);
170      AnchorPane.setRightAnchor(gameWindow, margin);
171
172    } catch (IOException e) {
173      logger.error(e.getMessage());
174    }
175  }
176}
