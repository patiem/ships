001package pl.korotkevics.ships.client.gui.util;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.InputStreamReader;
006import java.net.URL;
007import java.net.URLConnection;
008import java.util.Locale;
009import java.util.PropertyResourceBundle;
010import java.util.ResourceBundle;
011import java.util.ResourceBundle.Control;
012
013/**
014 * Used when instantiating ResourceBundle in order
015 * to enforce UTF-8 in UI (to make Polish, Russian
016 * special characters appear properly).
017 *
018 * <p> We're thankful to the author of this code.
019 * The author copied the default implementation and
020 * added "UTF-8" when instantiating InputStreamReader. </p>
021 *
022 * @author https://stackoverflow.com/questions/4659929/how-to-use-utf-8-in-resource-properties-with-resourcebundle
023 * @since 2011-01-11
024 */
025
026public class UTF8Control extends Control {
027
028  private static final String ENCODING = "UTF-8";
029
030  @Override
031  public ResourceBundle newBundle(final String baseName, final Locale locale, final String format,
032                                  final ClassLoader loader, final boolean reload)
033      throws IllegalAccessException, InstantiationException, IOException {
034    // The below is a copy of the default implementation.
035    String bundleName = toBundleName(baseName, locale);
036    String resourceName = toResourceName(bundleName, "properties");
037    ResourceBundle bundle = null;
038    InputStream stream = null;
039    if (reload) {
040      URL url = loader.getResource(resourceName);
041      if (url != null) {
042        URLConnection connection = url.openConnection();
043        if (connection != null) {
044          connection.setUseCaches(false);
045          stream = connection.getInputStream();
046        }
047      }
048    } else {
049      stream = loader.getResourceAsStream(resourceName);
050    }
051    if (stream != null) {
052      try {
053        // Only this line is changed to make it to read properties files as UTF-8.
054        bundle = new PropertyResourceBundle(new InputStreamReader(stream, ENCODING));
055      } finally {
056        stream.close();
057      }
058    }
059    return bundle;
060  }
061}
