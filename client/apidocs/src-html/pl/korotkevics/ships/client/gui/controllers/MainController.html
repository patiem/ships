001package pl.korotkevics.ships.client.gui.controllers;
002
003import javafx.event.ActionEvent;
004import javafx.fxml.FXML;
005import javafx.fxml.FXMLLoader;
006import javafx.scene.Parent;
007import javafx.scene.control.Button;
008import javafx.scene.control.Label;
009import javafx.scene.layout.Pane;
010import lombok.Getter;
011import pl.korotkevics.ships.client.client.Client;
012import pl.korotkevics.ships.client.localization.OurLocale;
013import pl.korotkevics.ships.shared.infra.logging.api.Target;
014import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
015
016import java.io.IOException;
017import java.util.Locale;
018import java.util.ResourceBundle;
019
020
021/**
022 * Main window (containing title and similar) controller.
023 *
024 * @author Magdalena Aarsman, Sandor Korotkevics
025 * @since 2017-12-14
026 */
027
028public class MainController {
029  
030  private static final Target logger = new SharedLogger(Client.class);
031  
032  private static final String DICTIONARY = "dict";
033  
034  @FXML
035  private Label gameTitle;
036  
037  @FXML
038  private Button russianButton;
039  
040  @FXML
041  private Pane mainPane;
042  
043  @Getter
044  private Client client;
045  
046  @FXML
047  private Button polishButton;
048  
049  @FXML
050  private Button englishButton;
051  
052  private ResourceBundle resourceBundle;
053  
054  /**
055   * Initialize main controller.
056   *
057   * @param client
058   *     - Instance of Client, which communicate with server.
059   */
060  @FXML
061  public void initialize(final Client client) {
062    this.client = client;
063    this.loadDefaultView();
064  }
065  
066  private void loadDefaultView() {
067    this.reloadView();
068  }
069  
070  private void reloadView() {
071    final FXMLLoader fxmlLoader = this.prepareFxmlLoader();
072    this.gameTitle.setText(this.resourceBundle.getString("gameTitle"));
073    try {
074      this.addToMainPane(fxmlLoader.load());
075    } catch (IOException e) {
076      logger.error(e.getMessage());
077    }
078  }
079  
080  private FXMLLoader prepareFxmlLoader() {
081    final String url = "/fxml/connectWindow.fxml";
082    final FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource(url));
083    this.resourceBundle = ResourceBundle.getBundle(DICTIONARY);
084    fxmlLoader.setResources(this.resourceBundle);
085    return fxmlLoader;
086  }
087  
088  private void addToMainPane(final Parent parent) {
089    this.mainPane.getChildren().clear();
090    this.mainPane.getChildren().add(parent);
091  }
092  
093  @FXML
094  void triggerPolishVersion(final ActionEvent event) {
095    logger.info("The language is set to " + OurLocale.POLISH);
096    Locale.setDefault(new Locale(OurLocale.POLISH.toString()));
097    this.reloadView();
098  }
099  
100  @FXML
101  void triggerEnglishVersion(final ActionEvent event) {
102    logger.info("The language is set to " + OurLocale.ENGLISH);
103    Locale.setDefault(new Locale(OurLocale.ENGLISH.toString()));
104    this.reloadView();
105  }
106  
107  @FXML
108  void triggerRussianVersion(final ActionEvent event) {
109    logger.info("The language is set to " + OurLocale.RUSSIAN);
110    Locale.setDefault(new Locale(OurLocale.RUSSIAN.toString()));
111    this.reloadView();
112  }
113  
114  void disableLocalizationButtons() {
115    this.hideButton(this.polishButton);
116    this.hideButton(this.englishButton);
117    this.hideButton(this.russianButton);
118  }
119  
120  private void hideButton(Button button) {
121    button.setDisable(true);
122    button.setVisible(false);
123  }
124}
