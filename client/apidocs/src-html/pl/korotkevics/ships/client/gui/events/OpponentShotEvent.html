001package pl.korotkevics.ships.client.gui.events;
002
003import javafx.event.Event;
004import javafx.event.EventType;
005import lombok.Getter;
006
007/**
008 * Event firing when opponent performs a shot.
009 *
010 * @author Magdalena Aarsman
011 * @since 2017-12-17
012 */
013public class OpponentShotEvent extends Event {
014  public static final EventType<OpponentShotEvent> OPPONENT_SHOT = new EventType<>(Event.ANY,
015                                                                                      "OPPONENT_SHOT");
016  
017  @Getter
018  final int shotIndex;
019  
020  public OpponentShotEvent(int shotIndex) {
021    super(OPPONENT_SHOT);
022    this.shotIndex = shotIndex;
023  }
024}
