001package pl.korotkevics.ships.client.validators;
002
003/**
004 * Enables validation of a port number.
005 *
006 * @author Magdalena Aarsman
007 * @since 2017-12-16
008 */
009public class PortValidator {
010
011  /**
012   * Validate the port number, returns port as Integer when it is valid.
013   *
014   * @param providedPort - port provided by user as a String
015   * @return - port number
016   * @throws IllegalArgumentException if port is invalid.
017   */
018  public int asInt(String providedPort) {
019    int port = Integer.parseInt(providedPort);
020    checkRange(port);
021    return port;
022  }
023
024  private void checkRange(int port) {
025    final int maxValidPort = 0xFFFF;
026    final int minValidPort = 1024;
027    if (port <= minValidPort || port > maxValidPort) {
028      throw new IllegalArgumentException("port value is out of range");
029    }
030  }
031}
