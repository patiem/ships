057  private static FieldState[] produceBoard(final List<Integer> occupiedPlaces) {
058    FieldState[] board = new FieldState[BOARD_SIZE * BOARD_SIZE];
059    for (int i = 0; i < BOARD_SIZE * BOARD_SIZE; i++) {
060      if (occupiedPlaces.contains(Integer.valueOf(i))) {
061        board[i] = FieldState.OCCUPIED;
062      } else {
063        board[i] = FieldState.EMPTY;
064      }
065    }
066
067    return board;
068  }
069
070  @Test(dataProvider = "validPlacement")
071  public void validPlacement(ShipOrientation shipOrientation, int shipStartIndex, int mastCount,
072                             FieldState[] board) {
073    //given
074    ShipPlacementValidator shipPlacementValidator =
075        new ShipPlacementValidator(shipOrientation, shipStartIndex, mastCount, board);
076    //when
077    boolean isValid = shipPlacementValidator.isPlacementValid();
078    //then
079    assertTrue(isValid);
080
081  }
082
083  @Test(dataProvider = "invalidPlacement")
084  void invalidPlacement(ShipOrientation shipOrientation, int shipStartIndex, int mastCount,
085                        FieldState[] board) {
086    //given
087    ShipPlacementValidator shipPlacementValidator =
088        new ShipPlacementValidator(shipOrientation, shipStartIndex, mastCount, board);
089    //when
090    boolean isValid = shipPlacementValidator.isPlacementValid();
091    //then
092    assertFalse(isValid);
093  }
094
095  @Test
096  void shipPassingToOtherSideShouldCauseInvalidPlacement() {
097    //given
098    ShipOrientation shipOrientation = ShipOrientation.VERTICAL;
099    int shipStartingIndex = 7;
100    int mastCount = 4;
101    FieldState[] board = produceBoard(Arrays.asList(1));
102    ShipPlacementValidator shipPlacementValidator =
103        new ShipPlacementValidator(shipOrientation, shipStartingIndex, mastCount, board);
104
105    //when
106    boolean isValid = shipPlacementValidator.isPlacementValid();
107
108    //then
109    assertFalse(isValid);
110  }
111
