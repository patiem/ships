86    private EventHandler<MouseEvent> onMouseEnteredOnShip =
87        event -> {
88          for (Node child : ((Group) event.getSource()).getChildren()) {
89            Rectangle rec = (Rectangle) child;
90            rec.setFill(Color.BLUE);
91          }
92          event.consume();
93        };
94  
95    private EventHandler<MouseEvent> onMouseExitFromShip =
96        event -> {
97          for (Node child : ((Group) event.getSource()).getChildren()) {
98            Rectangle rec = (Rectangle) child;
99            rec.setFill(Color.web("#7A16C2"));
100         }
101         event.consume();
102       };

104   private EventHandler<DragEvent> shipOnDragEntered =
105       event -> {
106         for (Node child : ((Group) event.getSource()).getChildren()) {
107           Rectangle rec = (Rectangle) child;
108           rec.setFill(Color.RED);
109         }
110         event.consume();
111       };

113   private EventHandler<MouseEvent> shipOnDragDetected =
114       event -> {
115         Dragboard db = ((Group) event.getSource()).startDragAndDrop(TransferMode.COPY);
116         ClipboardContent content = new ClipboardContent();
117         content.putString("Ship");
118         db.setContent(content);
119         event.consume();
120       };
121 
122   private EventHandler<DragEvent> shipOnDragDone =
123       event -> {
124         if (shipPlacementSuccess) {
125           for (Node child : ((Group) event.getSource()).getChildren()) {
126             Rectangle rec = (Rectangle) child;
127             rec.setFill(Color.GRAY);
128           }
129           ((Group) event.getSource()).setDisable(true);
130         }
131         shipPlacementSuccess = false;
132         event.consume();
133       };
134 
135   private EventHandler<DragEvent> boardOnDragOver =
136       event -> {
137         Dragboard db = event.getDragboard();
138         if (db.hasString()) {
139           event.acceptTransferModes(TransferMode.COPY);
140         }
141         event.consume();
142       };

144   @FXML

162   private void initializeBoard() {
163     final int margin = 50;
164     final NumberBinding allRectanglesWidth = Bindings.min(yourBoard.widthProperty(),
165         yourBoard.widthProperty().add(-margin));
166     final NumberBinding allRectanglesHeight = Bindings.min(yourBoard.heightProperty(),
167         yourBoard.widthProperty()).add(-margin);
168 
169     for (int i = 0; i < BOARD_SIZE; i++) {
170       for (int j = 0; j < BOARD_SIZE; j++) {
171         final int recIndex = j * BOARD_SIZE + i;
172         final int fillIndex = i * BOARD_SIZE + j;
173 
174         Rectangle yourRect = getYourRect(allRectanglesWidth, allRectanglesHeight);
175         yourBoard.add(yourRect, i, j);
176         setEventsOnField(yourRect, recIndex, fillIndex);
177         GridPane.setHalignment(yourRect, HPos.CENTER);
178       }
179     }
180   }

182   private Rectangle getYourRect(NumberBinding allRectanglesWidth,
183                                 NumberBinding allRectanglesHeight) {
184     final int initialSize = 15;
185     Rectangle yourRect = new Rectangle(initialSize, initialSize, Color.GRAY);
186     yourRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
187     yourRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
188     return yourRect;
189   }

191   private void setEventsOnField(Rectangle rectangle, final int recIndex, final int fillIndex) {
192     setFieldOnDragEntered(rectangle, fillIndex);
193     rectangle.setOnDragOver(boardOnDragOver);
194     setFieldOnDragExited(rectangle, fillIndex);
195     setFieldOnDragDropped(rectangle, recIndex, fillIndex);
196   }

198   private void setOpacity(int index, int mastCount, double opacity) {
199     index += 1;
200 
201     if (shipOrientation.equals(ShipOrientation.HORIZONTAL)) {
202       for (int i = 1; i < mastCount; i++) {
203         (yourBoard.getChildren().get(index + i * BOARD_SIZE)).setOpacity(opacity);
204       }
205     } else {
206       for (int i = 1; i < mastCount; i++) {
207         (yourBoard.getChildren().get(index + i)).setOpacity(opacity);
208       }
209     }
210   }

212   private void setFieldOnDragExited(Rectangle rectangle, int fillIndex) {
213     rectangle.setOnDragExited(
214         event -> {
215           double noOpacity = 1.0;
216           int mastCount = ((Group) event.getGestureSource()).getChildren().size();
217           if (isOutOfBound(fillIndex, mastCount)) {
218             return;
219           }
220           ((Rectangle) event.getSource()).setOpacity(noOpacity);
221           setOpacity(fillIndex, mastCount ,noOpacity);
222           event.consume();
223         });
224   }
225 
226   private void setFieldOnDragEntered(Rectangle rectangle, final int index) {
227     rectangle.setOnDragEntered(event -> {
228       double opacity = 0.5;
229       int mastCount = ((Group) event.getGestureSource()).getChildren().size();
230       if (shipOrientation.equals(ShipOrientation.HORIZONTAL) && isOutOfBound(index, mastCount)) {
231         return;
232       }
233       GridToBoardConverter gridToBoardConverter = new GridToBoardConverter(yourBoard);
234       ShipPlacementValidator shipPlacementValidator =
235           new ShipPlacementValidator(shipOrientation, index, mastCount,
236               gridToBoardConverter.convert());
237       if (!shipPlacementValidator.isPlacementValid()) {
238         return;
239       }
240 
241       ((Rectangle) event.getSource()).setOpacity(opacity);
242       setOpacity(index, mastCount, opacity);
243       event.consume();
244     });
245   }

247   private boolean isOutOfBound(int index, int mastCount) {
248     final int notRectangleChildCount = 2;
249     if (shipOrientation.equals(ShipOrientation.HORIZONTAL)) {
250       if (index + (mastCount - 1) * BOARD_SIZE
251           > yourBoard.getChildren().size() - notRectangleChildCount) {
252         return true;
253       }
254       return false;
255     } else {
256       if (index + mastCount >= yourBoard.getChildren().size()) {
257         return true;
258       }
259     }
260     return false;
261   }

263   private void setFieldOnDragDropped(Rectangle rectangle, final int recIndex, final int index) {
264     rectangle.setOnDragDropped(event -> {
265       Dragboard db = event.getDragboard();
266       boolean success = false;
267       if (db.hasString()) {
268         success = true;
269       }
270       event.setDropCompleted(success);
271       int mastCount = ((Group) event.getGestureSource()).getChildren().size();
272       Mast[] masts = new Mast[mastCount];
273 
274       if (shipOrientation.equals(ShipOrientation.HORIZONTAL) && isOutOfBound(index, mastCount)) {
275         shipPlacementSuccess = false;
276         return;
277       }
278 
279       GridToBoardConverter gridToBoardConverter = new GridToBoardConverter(yourBoard);
280       ShipPlacementValidator shipPlacementValidator =
281           new ShipPlacementValidator(shipOrientation, index, mastCount,
282               gridToBoardConverter.convert());
283       if (!shipPlacementValidator.isPlacementValid()) {
284         shipPlacementSuccess = false;
285         return;
286       }
287       masts[0] = Mast.ofIndex(String.valueOf(recIndex));
288       rectangle.setFill(Color.GREEN);
289       placeShip(index, masts, mastCount, recIndex);
290       ships.add(Ship.ofMasts(masts));
291       if (ships.size() == SHIPS_COUNT) {
292         buttonReady.setDisable(false);
293       }
294       shipPlacementSuccess = true;
295       event.consume();
296     });
297   }

299   private void placeShip(int index, Mast[] masts, int mastCount, int recIndex) {
300     index += 1;
301 
302     if (shipOrientation.equals(ShipOrientation.VERTICAL)) {
303       for (int i1 = 1; i1 < mastCount; i1++) {
304         ((Rectangle) yourBoard.getChildren().get(index + i1)).setFill(Color.GREEN);
305         masts[i1] = Mast.ofIndex(String.valueOf(recIndex + i1 * BOARD_SIZE));
306       }
307     } else {
308       for (int i1 = 1; i1 < mastCount; i1++) {
309         ((Rectangle) yourBoard.getChildren().get(index + i1 * BOARD_SIZE)).setFill(Color.GREEN);
310         masts[i1] = Mast.ofIndex(String.valueOf(recIndex + i1));
311       }
312     }
313   }

322   private void setEventsOnShips(Group shipsGroup) {
323     for (Node ship : shipsGroup.getChildren()) {
324       ship.setOnMouseEntered(onMouseEnteredOnShip);
325       ship.setOnMouseExited(onMouseExitFromShip);
326       ship.setOnDragEntered(shipOnDragEntered);
327       ship.setOnDragDetected(shipOnDragDetected);
328       ship.setOnDragDone(shipOnDragDone);
329     }

341   private void loadGameWindow() {
342     try {
343       getClient().sendFleet(Fleet.ofShips(ships));
344 
345       final String gameWindowUrl = "/fxml/gameWindow.fxml";
346       final FXMLLoader gameWindowLoader = new FXMLLoader(getClass().getResource(gameWindowUrl));
347       final Parent gameWindow = gameWindowLoader.load();
348       final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
349 
350       Rectangle2D screenBounds = Screen.getPrimary().getVisualBounds();
351 
352       Stage stage = (Stage) mainPane.getScene().getWindow();
353       final int shrinkSize = 300;
354       stage.setMinHeight(screenBounds.getHeight() - shrinkSize);
355       stage.setMinWidth(screenBounds.getWidth() - shrinkSize);
356 
357       mainPane.getChildren().clear();
358       mainPane.getChildren().setAll(gameWindow);
359 
360       GameController gameController = gameWindowLoader.getController();
361       gameController.initializeClient();
362       gameController.initializeBoards(yourBoard);
363 
364       final double margin = 0.0;
365       AnchorPane.setTopAnchor(gameWindow, margin);
366       AnchorPane.setBottomAnchor(gameWindow, margin);
367       AnchorPane.setLeftAnchor(gameWindow, margin);
368       AnchorPane.setRightAnchor(gameWindow, margin);
369 
370     } catch (IOException e) {
371       logger.error(e.getMessage());
372     }
