1   package com.epam.ships.client.client;
2   
3   import com.epam.ships.infra.communication.api.message.Header;
4   
5   import java.util.EnumMap;
6   import java.util.Map;
7   
8   /**
9    * Helper class to build MessageHandler.
10   * @author Magdalena Aarsman
11   * @since 2018-01-02
12   */
13  
14  public class MessageHandlerBuilder {
15    private Map<Header, EventTrigger> triggers;
16  
17    /**
18     * Set triggers map to be enum map with Header as key.
19     * @return MessageHandlerBuilder
20     */
21    public MessageHandlerBuilder withEnumMap() {
22      triggers = new EnumMap<>(Header.class);
23      return this;
24    }
25  
26    /**
27     * Set default set of triggers and put it into map.
28     * @return MessageHandlerBuilder
29     */
30    public MessageHandlerBuilder withDefaultSetsOfTriggers() {
31      triggers.put(Header.OPPONENT_CONNECTED, new OpponentConnectedTrigger());
32      triggers.put(Header.SHOT, new OpponentShotTrigger());
33      triggers.put(Header.CONNECTION, new ConnectionEndTrigger());
34      triggers.put(Header.YOUR_TURN, new TurnTrigger());
35      triggers.put(Header.MISS, new MissShotTrigger());
36      triggers.put(Header.HIT, new HitShotTrigger());
37      triggers.put(Header.SHIP_DESTRUCTED, new HitShotTrigger());
38      triggers.put(Header.WIN, new WinTrigger());
39      triggers.put(Header.LOSE, new LoseTrigger());
40      return this;
41    }
42  
43    /**
44     * Build MessageHandler.
45     * @return MessageHandler
46     */
47    public MessageHandler build() {
48      return new MessageHandler(triggers);
49    }
50  }
