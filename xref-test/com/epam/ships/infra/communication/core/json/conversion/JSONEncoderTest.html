20          .withAuthor(Author.SERVER)
21          .withHeader(Header.CONNECTION)
22          .withStatus(Status.OK)
23          .withStatement("The first real test in this project:-)")
24          .build();

28      assertEquals(jsonElement.toString(), "{\"header\":\"CONNECTION\",\"status\":\"OK\",\"author\":\"SERVER\"," +
29          "\"statement\":\"The first real test in this project:-)\"," +
30          "\"fleet\":{\"fleet\":{}}}");
31    }
