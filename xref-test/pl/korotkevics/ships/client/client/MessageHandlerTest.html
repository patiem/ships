22  public class MessageHandlerTest {
23  
24    @BeforeClass
25    public void setUp() throws InterruptedException {
26      if(!JavaFxInitializer.isLaunched()) {
27        JavaFxInitializer.initialize();
28      }
29    }
30  
31    @BeforeMethod
32    public void skipTestIfThereIsNoGraphicsSupport() {
33      if (!JavaFxInitializer.isEnable()){
34        throw new SkipException("skipping test because of lack of support for graphics");
35      }
36    }
37  
38    @Test
39    public void shouldCallConnectionEndTrigger() {
40      //given
41      ConnectionEndTrigger connectionEndTrigger = mock(ConnectionEndTrigger.class);
42      MessageHandler messageHandler = produceMessageHandler(connectionEndTrigger);
43      Message message = connectionEndMessage();
44  
45      //when
46      messageHandler.handle(message);
47  
48      //then
49      verify(connectionEndTrigger, times(1)).fire(messageHandler.getCurrentEventButton(), message.getStatement());
50    }
51  
52    @Test
53    public void shouldReturnTrueAfterTriggerOpponentWithdrawEven() {
54      //given
55      ConnectionEndTrigger connectionEndTrigger = mock(ConnectionEndTrigger.class);
56      MessageHandler messageHandler = produceMessageHandler(connectionEndTrigger);
57      Message message = connectionEndMessage();
58  
59      //when
60      messageHandler.handle(message);
61  
62      assertTrue(messageHandler.isEndConnectionTriggered());
63    }
64  
65    private Message connectionEndMessage() {
66      return new MessageBuilder()
67          .withHeader(Header.CONNECTION)
68          .withStatus(Status.END)
69          .withStatement("End of a message")
70          .build();
71    }
72  
73    private MessageHandler produceMessageHandler(EventTrigger eventTrigger) {
74      Map<Header, EventTrigger> triggers = new EnumMap<>(Header.class);
75      triggers.put(Header.CONNECTION, eventTrigger);
76      MessageHandler messageHandler = new MessageHandler(triggers);
77      messageHandler.setCurrentEventButton(new Button());
78      return messageHandler;
79    }
